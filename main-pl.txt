\title{\textbf{O rozmiarze zbiorów semi-rozcinających w grafach}}
\author{}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Niech $G=(V,E)$ będzie skończonym grafem nieskierowanym, $T\subseteq V$ oraz $F\subseteq E$. Zakładamy, że po usunięciu
krawędzi z~$F$ odległość między każdą parą różnych wierzchołków z~$T$ ściśle rośnie. W pracy podajemy samowystarczalny dowód
nierówności $|F|\ge |T|-1$. Dowód składa się z~trzech kroków. Najpierw zapisujemy warunek semi-rozcinania w~języku przecinania
najkrótszych ścieżek. Następnie dowodzimy, że własność ta zachowuje się przy ściąganiu domkniętego sąsiedztwa terminala
nieincydentnego z~$F$. Na końcu stosujemy indukcję po parze $(|T|,|V|)$ w~porządku leksykograficznym.
\end{abstract}

\section{Wprowadzenie}

Rozpatrujemy następujący problem strukturalny. Dany jest graf $G=(V,E)$, zbiór wyróżnionych wierzchołków $T\subseteq V$ oraz
zbiór krawędzi $F\subseteq E$. Interesuje nas sytuacja, w~której usunięcie $F$ powoduje ścisły wzrost odległości między każdą
parą różnych wierzchołków z~$T$. Celem jest wyznaczenie uniwersalnego dolnego ograniczenia na rozmiar takiego zbioru krawędzi.
Główny wynik stwierdza, że zawsze zachodzi nierówność $|F|\ge |T|-1$. Dowód nie zakłada żadnych dodatkowych własności grafu,
w~szczególności nie zakładamy spójności.

\section{Notacja i definicje}

W całym tekście $G=(V,E)$ oznacza skończony graf nieskierowany prosty. Każdą krawędź utożsamiamy z~dwuelementowym podzbiorem
zbioru $V$. Jeżeli $e\in E$ oraz $v\in e$, to mówimy, że $v$ jest końcem krawędzi $e$. Odległości traktujemy jako elementy
zbioru $\mathbb{N}_0\cup\{\infty\}$ z~naturalnym porządkiem rozszerzonym o~relacje $n<\infty$ dla każdego $n\in\mathbb{N}_0$.

\begin{definition}[Spacery, ścieżki i odległość]\label{def:paths-distance}
Niech $u,w\in V$. Spacerem z~$u$ do~$w$ nazywamy ciąg wierzchołków
\[
  P=(v_0,v_1,\dots,v_k),
\]
taki że $v_0=u$, $v_k=w$ oraz $\{v_{i-1},v_i\}\in E$ dla każdego $i\in\{1,\dots,k\}$. Długość spaceru $P$ definiujemy przez
$\len{P}\coloneqq k$.

Ścieżką nazywamy spacer $P=(v_0,\dots,v_k)$, dla którego wierzchołki $v_0,\dots,v_k$ są parami różne. Dla ścieżki
$P=(v_0,\dots,v_k)$ oznaczamy
\[
  V(P)\coloneqq \{v_0,\dots,v_k\},
  \qquad
  E(P)\coloneqq \bigl\{\{v_{i-1},v_i\}: i=1,\dots,k\bigr\}.
\]

Odległość $\dist{G}{u}{w}$ definiujemy jako minimum długości ścieżek z~$u$ do~$w$ w~$G$, jeżeli taka ścieżka istnieje; w przeciwnym
razie kładziemy $\dist{G}{u}{w}\coloneqq \infty$.
\end{definition}

\begin{notation}[Podstawowe operacje na grafie]\label{not:basic-operations}
Jeżeli $F\subseteq E$, to przez $G\setminus F$ oznaczamy graf $(V,E\setminus F)$. Dla $F\subseteq E$ piszemy
\[
  V[F]\coloneqq \{v\in V:\ \exists\, e\in F\ \text{ takie, że }\ v\in e\}.
\]
Jeżeli $v\in V$, to jego otoczenie i~domknięte otoczenie oznaczamy odpowiednio przez
\[
  N(v)\coloneqq \{w\in V:\ \{v,w\}\in E\},
  \qquad
  N[v]\coloneqq N(v)\cup\{v\}.
\]
\end{notation}

\begin{definition}[Ściąganie zbioru wierzchołków]\label{def:contraction}
Niech $W\subseteq V$ oraz niech $w^\ast\notin V$ będzie nowym wierzchołkiem. Definiujemy odwzorowanie
\[
  \pi_W\colon V\to (V\setminus W)\cup\{w^\ast\},
  \qquad
  \pi_W(v)\coloneqq
  \begin{cases}
    w^\ast & \text{dla } v\in W,\\
    v & \text{dla } v\notin W.
  \end{cases}
\]
Przez $G/W$ oznaczamy graf $(V',E')$, gdzie
\[
  V'\coloneqq (V\setminus W)\cup\{w^\ast\},
  \qquad
  E'\coloneqq \bigl\{\{\pi_W(x),\pi_W(y)\}: \{x,y\}\in E,\ \pi_W(x)\neq \pi_W(y)\bigr\}.
\]
Jeżeli $e=\{x,y\}\in E$ oraz $\pi_W(x)\neq \pi_W(y)$, to obraz tej krawędzi oznaczamy przez
\[
  e/W\coloneqq \{\pi_W(x),\pi_W(y)\}\in E'.
\]
Dla $F\subseteq E$ definiujemy obraz zbioru krawędzi przez
\[
  F/W\coloneqq \{e/W:\ e\in F,\ e/W\in E'\}\subseteq E'.
\]
Odwzorowanie $e\mapsto e/W$, określone na zbiorze tych krawędzi z~$F$, które mają co najwyżej jeden koniec w~$W$, jest surjekcją
na $F/W$. W~szczególności
\[
  |F/W|\le |F|.
\]
\end{definition}

\begin{definition}[Zbiór semi-rozcinający]\label{def:semi-cut}
Niech $T\subseteq V$. Mówimy, że $F\subseteq E$ jest zbiorem semi-rozcinającym dla~$T$ w~$G$, jeżeli dla każdej pary różnych
wierzchołków $u,w\in T$ zachodzi
\[
  \dist{G}{u}{w}<\dist{G\setminus F}{u}{w}.
\]
\end{definition}

\section{Lematy o najkrótszych ścieżkach}

\begin{lemma}\label{lem:walk-to-path}
Jeżeli istnieje spacer z~$u$ do~$w$ w~$G$, to istnieje również ścieżka z~$u$ do~$w$ w~$G$ o~długości nie większej niż długość tego
spaceru.
\end{lemma}

\begin{proof}
Niech $Q$ będzie ustalonym spacerem z~$u$ do~$w$ w~$G$. Wśród wszystkich spacerów z~$u$ do~$w$ o~długości nie większej niż $\len{Q}$
wybierzmy spacer $P=(v_0,\dots,v_k)$ o~najmniejszej długości. Wystarczy wykazać, że $P$ jest ścieżką.

Przypuśćmy przeciwnie, że istnieją indeksy $0\le i<j\le k$ takie, że $v_i=v_j$. Jeżeli $j<k$, to z~$\{v_j,v_{j+1}\}\in E$ oraz
$v_i=v_j$ wynika $\{v_i,v_{j+1}\}\in E$. W~obu przypadkach, to znaczy zarówno dla $j<k$, jak i~dla $j=k$, ciąg
\[
  P'=(v_0,\dots,v_i,v_{j+1},\dots,v_k)
\]
jest spacerem z~$u$ do~$w$ o~długości mniejszej niż $\len{P}$. Otrzymujemy sprzeczność z~minimalnością wyboru spaceru $P$.
Zatem $P$ nie ma powtórzeń wierzchołków, czyli jest ścieżką.
\end{proof}

\begin{lemma}\label{lem:subpath}
Niech $P=(v_0,\dots,v_k)$ będzie ścieżką z~$u$ do~$w$ w~$G$ o~długości $\len{P}=\dist{G}{u}{w}$. Dla indeksów
$0\le i\le j\le k$ oznaczmy przez
\[
  P[v_i,v_j]\coloneqq (v_i,\dots,v_j)
\]
podścieżkę $P$ między wierzchołkami $v_i$ i~$v_j$. Wówczas
\[
  \len{P[v_i,v_j]}=\dist{G}{v_i}{v_j}.
\]
\end{lemma}

\begin{proof}
Załóżmy przeciwnie, że istnieje ścieżka $Q$ z~$v_i$ do~$v_j$ taka, że
\[
  \len{Q}<\len{P[v_i,v_j]}.
\]
Wtedy ciąg
\[
  (v_0,\dots,v_i)\cdot Q\cdot (v_j,\dots,v_k)
\]
jest spacerem z~$u$ do~$w$ w~$G$ o~długości
\[
  i+\len{Q}+(k-j)
  <
  i+\len{P[v_i,v_j]}+(k-j)
  =
  \len{P}
  =
  \dist{G}{u}{w},
\]
co przeczy definicji odległości. Zatem podścieżka $P[v_i,v_j]$ jest ścieżką najkrótszą między swoimi końcami.
\end{proof}

\begin{observation}\label{obs:shortest-paths}
Niech $T\subseteq V$ oraz $F\subseteq E$. Wówczas $F$ jest zbiorem semi-rozcinającym dla~$T$ w~$G$ wtedy i~tylko wtedy, gdy dla
każdych różnych $u,w\in T$ zachodzą jednocześnie następujące warunki:
\begin{enumerate}
  \item $\dist{G}{u}{w}<\infty$;
  \item każda ścieżka $P$ z~$u$ do~$w$ o~długości $\len{P}=\dist{G}{u}{w}$ spełnia $E(P)\cap F\neq\emptyset$.
\end{enumerate}
\end{observation}

\begin{proof}
Ustalmy różne wierzchołki $u,w\in T$.

Załóżmy najpierw, że $F$ jest zbiorem semi-rozcinającym dla~$T$. Wtedy
\[
  \dist{G}{u}{w}<\dist{G\setminus F}{u}{w}\le \infty,
\]
więc $\dist{G}{u}{w}<\infty$. Niech $P$ będzie dowolną ścieżką z~$u$ do~$w$ o~długości $\dist{G}{u}{w}$. Gdyby
$E(P)\cap F=\emptyset$, to $P$ byłaby także ścieżką w~$G\setminus F$, a~zatem
\[
  \dist{G\setminus F}{u}{w}\le \len{P}=\dist{G}{u}{w},
\]
co przeczy założeniu.

Załóżmy teraz, że spełnione są warunki (1) i~(2). Jeżeli w~$G\setminus F$ nie istnieje ścieżka z~$u$ do~$w$, to
$\dist{G\setminus F}{u}{w}=\infty$ i~teza jest natychmiastowa. Pozostaje przypadek $\dist{G\setminus F}{u}{w}<\infty$.
Niech $Q$ będzie ścieżką z~$u$ do~$w$ w~$G\setminus F$ o~długości
\[
  \len{Q}=\dist{G\setminus F}{u}{w}.
\]
Ponieważ $Q$ jest także ścieżką w~$G$, mamy $\len{Q}\ge \dist{G}{u}{w}$. Równość nie może zachodzić, gdyż wtedy $Q$ byłaby ścieżką
najkrótszą w~$G$ nieprzecinającą zbioru $F$, wbrew warunkowi (2). Dostajemy więc
\[
  \dist{G\setminus F}{u}{w}=\len{Q}>\dist{G}{u}{w}.
\]
Ponieważ para $u,w$ była dowolna, zbiór $F$ jest semi-rozcinający dla~$T$.
\end{proof}

\begin{lemma}\label{lem:projection-walk}
Niech $W\subseteq V$. Jeżeli $P=(v_0,\dots,v_k)$ jest spacerem w~$G$, to w~$G/W$ istnieje spacer
\[
  P'=(w_0,\dots,w_m)
\]
taki, że $w_0=\pi_W(v_0)$, $w_m=\pi_W(v_k)$ oraz $m\le k$.
\end{lemma}

\begin{proof}
Niech $a_i\coloneqq \pi_W(v_i)$ dla $i=0,\dots,k$. Niech
\[
  0=i_0<i_1<\dots<i_m\le k
\]
będą pierwszymi indeksami kolejnych maksymalnych bloków stałych w~ciągu $(a_0,\dots,a_k)$. Innymi słowy, dla każdego
$r\in\{0,\dots,m-1\}$ zachodzi
\[
  a_{i_r}=a_{i_r+1}=\dots=a_{i_{r+1}-1}
  \qquad\text{oraz}\qquad
  a_{i_r}\neq a_{i_{r+1}},
\]
a ponadto $a_{i_m}=a_k$. Definiujemy
\[
  w_r\coloneqq a_{i_r}
  \qquad\text{dla } r=0,\dots,m.
\]
Wtedy $w_0=a_0=\pi_W(v_0)$, $w_m=a_k=\pi_W(v_k)$ oraz oczywiście $m\le k$.

Pozostaje sprawdzić, że kolejne wierzchołki $w_r,w_{r+1}$ są połączone krawędzią w~$G/W$. Dla
$r\in\{0,\dots,m-1\}$ mamy
\[
  a_{i_{r+1}-1}=a_{i_r}
  \qquad\text{oraz}\qquad
  a_{i_{r+1}-1}\neq a_{i_{r+1}}.
\]
Ponieważ $\{v_{i_{r+1}-1},v_{i_{r+1}}\}\in E$, z~Definicji~\ref{def:contraction} wynika, że
\[
  \{w_r,w_{r+1}\}
  =
  \{\pi_W(v_{i_{r+1}-1}),\pi_W(v_{i_{r+1}})\}
  \in E(G/W).
\]
Zatem $P'$ jest spacerem w~$G/W$.
\end{proof}

\section{Zachowanie własności przy ściąganiu}

\begin{lemma}\label{lem:lifting-endpoint}
Niech $t\in V$, niech $t^\prime\notin V$ będzie nowym wierzchołkiem oraz niech $G^\prime\coloneqq G/N[t]$, gdzie zbiór $N[t]$ został
ściągnięty do~$t^\prime$. Niech $z\in V\setminus N[t]$ i~niech
\[
  P^\prime=(t^\prime,x_1,\dots,x_k)
\]
będzie ścieżką najkrótszą z~$t^\prime$ do~$z$ w~$G^\prime$. Wówczas istnieje wierzchołek $v\in N(t)$ taki, że ciąg
\[
  P=(t,v,x_1,\dots,x_k)
\]
jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.
\end{lemma}

\begin{proof}
Ponieważ $z\neq t^\prime$, mamy $k\ge 1$. Krawędź $\{t^\prime,x_1\}$ należy do~$E(G^\prime)$, więc z~Definicji~\ref{def:contraction}
wynika istnienie krawędzi $\{v,x_1\}\in E(G)$ takiej, że $v\in N[t]$ oraz $\{v,x_1\}/N[t]=\{t^\prime,x_1\}$. Z~założenia
$x_1\notin N[t]$, a~zatem $v\neq t$; w~konsekwencji $v\in N(t)$.

Pokażemy, że
\[
  P=(t,v,x_1,\dots,x_k)
\]
jest ścieżką w~$G$. Wierzchołek $v$ jest sąsiadem $t$, więc $\{t,v\}\in E$. Dla każdego $i\in\{1,\dots,k\}$ mamy
$x_i\notin N[t]$, zatem $x_i\neq t$ oraz $x_i\neq v$. Ponieważ $P^\prime$ jest ścieżką, wierzchołki $x_1,\dots,x_k$ są parami różne.
Wobec tego $P$ jest ścieżką i~spełnia
\[
  \len{P}=\len{P^\prime}+1.
\]

Niech teraz $Q=(q_0,\dots,q_\ell)$ będzie dowolną ścieżką z~$t$ do~$z$ w~$G$. Ponieważ $z\notin N[t]$ oraz $q_0=t$, mamy $\ell\ge 2$
i~$q_1\in N(t)$. W szczególności
\[
  \pi_{N[t]}(q_0)=\pi_{N[t]}(q_1)=t^\prime.
\]
Stosując Lemat~\ref{lem:projection-walk} do ścieżki $Q$, otrzymujemy w~$G^\prime$ spacer z~$t^\prime$ do~$z$ o~długości co najwyżej
$\ell-1$. Z definicji odległości dostajemy
\[
  \dist{G^\prime}{t^\prime}{z}\le \ell-1=\len{Q}-1.
\]
Ponieważ $Q$ było dowolną ścieżką z~$t$ do~$z$, z ostatniej nierówności wynika
\[
  \dist{G}{t}{z}\ge \dist{G^\prime}{t^\prime}{z}+1.
\]
Z drugiej strony $P$ jest ścieżką z~$t$ do~$z$, więc
\[
  \dist{G}{t}{z}\le \len{P}=\len{P^\prime}+1=\dist{G^\prime}{t^\prime}{z}+1.
\]
Obie nierówności dają równość
\[
  \dist{G}{t}{z}=\len{P},
\]
czyli $P$ jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.
\end{proof}

\begin{proposition}\label{prop:contraction}
Niech $T\subseteq V$ oraz $F\subseteq E$. Załóżmy, że $F$ jest zbiorem semi-rozcinającym dla~$T$ w~$G$. Niech
$t\in T\setminus V[F]$ oraz niech $t^\prime\notin V$ będzie nowym wierzchołkiem. Oznaczmy
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
Wówczas $F^\prime$ jest zbiorem semi-rozcinającym dla~$T^\prime$ w~$G^\prime$.
\end{proposition}

\begin{proof}
Najpierw pokażemy, że
\[
  T\setminus\{t\}\subseteq V\setminus N[t].
\]
Weźmy dowolne $u\in T\setminus\{t\}$ i~przypuśćmy, że $u\in N(t)$. Wtedy $\dist{G}{t}{u}=1$, a~jedyna ścieżka długości $1$ z~$t$
do~$u$ jest równa $(t,u)$. Z~Obserwacji~\ref{obs:shortest-paths} wynikałoby, że krawędź $\{t,u\}$ należy do~$F$, co przeczy
założeniu $t\notin V[F]$. Otrzymana sprzeczność dowodzi, że żaden terminal różny od~$t$ nie należy do~$N[t]$. W szczególności
$T^\prime\subseteq V(G^\prime)$.

Pokażemy teraz, że dla każdych różnych $u,w\in T^\prime$ zachodzi $\dist{G^\prime}{u}{w}<\infty$. Jeżeli
$u,w\in T\setminus\{t\}$, to z~Definicji~\ref{def:semi-cut} wynika $\dist{G}{u}{w}<\infty$, więc w~$G$ istnieje ścieżka
z~$u$ do~$w$. Z~Lematu~\ref{lem:projection-walk} otrzymujemy spacer z~$u$ do~$w$ w~$G^\prime$, a następnie z~Lematu~\ref{lem:walk-to-path}
ścieżkę z~$u$ do~$w$ w~$G^\prime$. Jeżeli jeden z~wierzchołków $u,w$ jest równy $t^\prime$, to drugi należy do~$T\setminus\{t\}$;
oznaczmy go przez $z$. Wtedy z~Definicji~\ref{def:semi-cut} mamy $\dist{G}{t}{z}<\infty$, więc w~$G$ istnieje ścieżka z~$t$ do~$z$.
Jej obraz daje, po użyciu lematów \ref{lem:projection-walk} i~\ref{lem:walk-to-path}, ścieżkę z~$t^\prime$ do~$z$ w~$G^\prime$.

Niech teraz $u,w\in T^\prime$ będą różne i~niech $P^\prime$ będzie dowolną ścieżką najkrótszą z~$u$ do~$w$ w~$G^\prime$. Zgodnie
z~Obserwacją~\ref{obs:shortest-paths} wystarczy wykazać, że
\[
  E(P^\prime)\cap F^\prime\neq\emptyset.
\]
Rozpatrzymy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $t^\prime\notin V(P^\prime)$.}
Wszystkie wierzchołki ścieżki $P^\prime$ należą wtedy do~$V\setminus N[t]$. Każda krawędź ścieżki $P^\prime$ ma więc oba końce
poza $N[t]$, a~z definicji ściągania wynika, że jest to zarazem krawędź grafu $G$. W konsekwencji $P^\prime$ jest ścieżką z~$u$ do~$w$
w~$G$.

Pokażemy, że $P^\prime$ jest najkrótsza również w~$G$. Gdyby istniała w~$G$ ścieżka $Q$ z~$u$ do~$w$ o~długości
$\len{Q}<\len{P^\prime}$, to na mocy Lematu~\ref{lem:projection-walk} w~$G^\prime$ istniałby spacer z~$u$ do~$w$ o~długości
co najwyżej $\len{Q}$. Z~Lematu~\ref{lem:walk-to-path} wynikałoby więc, że w~$G^\prime$ istnieje ścieżka z~$u$ do~$w$ o~długości
co najwyżej $\len{Q}<\len{P^\prime}$, sprzecznie z~wyborem $P^\prime$. Otrzymujemy
\[
  \len{P^\prime}=\dist{G}{u}{w}.
\]
Ponieważ $u,w\in T\setminus\{t\}$, z~Obserwacji~\ref{obs:shortest-paths} dostajemy krawędź $e\in E(P^\prime)\cap F$. Oba końce $e$
leżą poza $N[t]$, więc $e/N[t]=e$ oraz $e\in F^\prime$. Zatem $E(P^\prime)\cap F^\prime\neq\emptyset$.

\medskip\noindent
\textbf{Przypadek 2: $t^\prime\in V(P^\prime)$.}
Jeżeli $t^\prime$ jest ostatnim wierzchołkiem ścieżki $P^\prime$, zastępujemy $P^\prime$ ścieżką przeciwną. Nie zmienia to ani długości,
ani zbioru krawędzi. Możemy więc zapisać
\[
  P^\prime=(y_0,\dots,y_r,y_{r+1},\dots,y_m),
  \qquad
  y_r=t^\prime,
\]
gdzie $r<m$. Oznaczmy przez
\[
  Q^\prime\coloneqq (t^\prime,y_{r+1},\dots,y_m)
\]
sufiks ścieżki $P^\prime$ zaczynający się w~$t^\prime$, oraz połóżmy $z\coloneqq y_m$. Wierzchołek $z$ należy do
$T^\prime\setminus\{t^\prime\}=T\setminus\{t\}$, a~więc $z\notin N[t]$.

Jeżeli $r=0$, to $Q^\prime=P^\prime$, więc $Q^\prime$ jest ścieżką najkrótszą z~$t^\prime$ do~$z$ w~$G^\prime$. Jeżeli $r>0$, to
z~Lematu~\ref{lem:subpath} wynika, że $Q^\prime$ jest ścieżką najkrótszą między swoimi końcami. W~obu sytuacjach możemy zastosować
Lemat~\ref{lem:lifting-endpoint}. Dostajemy wierzchołek $v\in N(t)$ taki, że
\[
  Q=(t,v,y_{r+1},\dots,y_m)
\]
jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.

Z~Obserwacji~\ref{obs:shortest-paths} wynika, że $E(Q)\cap F\neq\emptyset$. Pierwsza krawędź ścieżki $Q$ jest równa $\{t,v\}$, więc
nie należy do~$F$, gdyż $t\notin V[F]$. Zatem istnieje krawędź $e\in E(Q)\cap F$ różna od~$\{t,v\}$. Taka krawędź leży na odcinku
$(v,y_{r+1},\dots,y_m)$, a~jej obraz po ściągnięciu $N[t]$ należy do~$E(Q^\prime)\subseteq E(P^\prime)$. Ponieważ $e\in F$, mamy także
$e/N[t]\in F^\prime$. Stąd ponownie
\[
  E(P^\prime)\cap F^\prime\neq\emptyset.
\]

W obu przypadkach każda ścieżka najkrótsza między różnymi wierzchołkami z~$T^\prime$ przecina zbiór $F^\prime$. Z~Obserwacji
\ref{obs:shortest-paths} wynika, że $F^\prime$ jest zbiorem semi-rozcinającym dla~$T^\prime$ w~$G^\prime$.
\end{proof}

\section{Główne twierdzenie}

\begin{theorem}\label{thm:main}
Niech $G=(V,E)$ będzie grafem, niech $T\subseteq V$ oraz $F\subseteq E$. Jeżeli $F$ jest zbiorem semi-rozcinającym dla~$T$ w~$G$, to
\[
  |F|\ge |T|-1.
\]
\end{theorem}

\begin{proof}
Dla par liczb całkowitych nieujemnych $(k,n)$ oznaczmy przez $\mathcal{P}(k,n)$ następujące zdanie:
\begin{quote}
dla każdego grafu $H$ o~$n$ wierzchołkach, każdego zbioru $X\subseteq V(H)$ o~mocy $k$ oraz każdego zbioru krawędzi
$M\subseteq E(H)$, jeżeli $M$ jest zbiorem semi-rozcinającym dla~$X$ w~$H$, to $|M|\ge k-1$.
\end{quote}
Udowodnimy prawdziwość wszystkich zdań $\mathcal{P}(k,n)$ przez indukcję względem porządku leksykograficznego na parach $(k,n)$.

\medskip\noindent
\textbf{Baza indukcji.} Niech $k\le 1$. Wtedy dla dowolnego grafu $H$, dowolnego $X\subseteq V(H)$ o~mocy $k$ oraz dowolnego
$M\subseteq E(H)$ zachodzi nierówność
\[
  |M|\ge 0\ge k-1.
\]
Zatem $\mathcal{P}(k,n)$ jest prawdziwe dla każdego $n$.

\medskip\noindent
\textbf{Krok indukcyjny.} Ustalmy parę $(k,n)$ z~$k\ge 2$ i~załóżmy, że $\mathcal{P}(k',n')$ jest prawdziwe dla każdej pary
$(k',n')$ mniejszej leksykograficznie od~$(k,n)$. Rozważmy graf $G=(V,E)$ o~$n$ wierzchołkach, zbiór $T\subseteq V$ o~mocy $k$
oraz zbiór $F\subseteq E$, który jest semi-rozcinający dla~$T$ w~$G$. Pokażemy, że $|F|\ge k-1$.

Rozpatrzymy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $T\cap V[F]=\emptyset$.}
Wybierzmy $t\in T$ oraz $u\in T\setminus\{t\}$. Z~Definicji~\ref{def:semi-cut} wynika, że $\dist{G}{t}{u}<\infty$. Istnieje więc
ścieżka z~$t$ do~$u$ o~dodatniej długości, a~zatem $N(t)\neq\emptyset$ i~w szczególności
\[
  |N[t]|\ge 2.
\]

Niech
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
Na mocy Propozycji~\ref{prop:contraction} zbiór $F^\prime$ jest semi-rozcinający dla~$T^\prime$ w~$G^\prime$. Ponadto
\[
  |T^\prime|=|T|=k
  \qquad\text{oraz}\qquad
  |V(G^\prime)|=|V|-|N[t]|+1<n.
\]
Para odpowiadająca trójce $(G^\prime,T^\prime,F^\prime)$ jest więc równa $(k,n')$ dla pewnego $n'<n$. Z~założenia indukcyjnego
otrzymujemy
\[
  |F^\prime|\ge |T^\prime|-1=k-1.
\]
Z~Definicji~\ref{def:contraction} wiemy, że $|F^\prime|\le |F|$, a~więc
\[
  |F|\ge k-1.
\]

\medskip\noindent
\textbf{Przypadek 2: $T\cap V[F]\neq\emptyset$.}
Wybierzmy wierzchołek $t\in T\cap V[F]$ oraz krawędź $e\in F$ taką, że $t\in e$. Oznaczmy
\[
  T^-\coloneqq T\setminus\{t\},
  \qquad
  F^-\coloneqq F\setminus\{e\}.
\]
Pokażemy, że $F^-$ jest zbiorem semi-rozcinającym dla~$T^-$ w~$G$.

Weźmy dowolne różne wierzchołki $u,w\in T^-$. Z~Definicji~\ref{def:semi-cut} wynika, że $\dist{G}{u}{w}<\infty$. Niech $P$ będzie
dowolną ścieżką najkrótszą z~$u$ do~$w$ w~$G$. Z~Obserwacji~\ref{obs:shortest-paths} dostajemy
\[
  E(P)\cap F\neq\emptyset.
\]
Jeżeli $e\notin E(P)$, to natychmiast
\[
  E(P)\cap F^-\neq\emptyset.
\]
Pozostaje przypadek $e\in E(P)$. Ponieważ $t\in e$, wierzchołek $t$ należy do~$V(P)$. Zapiszmy $P$ w~postaci
\[
  P=P_u\cdot P_w,
\]
gdzie $P_u$ jest podścieżką od~$u$ do~$t$, a~$P_w$ podścieżką od~$t$ do~$w$. Z~Lematu~\ref{lem:subpath} wynika, że
$P_u$ i~$P_w$ są ścieżkami najkrótszymi odpowiednio między parami $(u,t)$ i~$(t,w)$.

Ponieważ $P$ jest ścieżką, podścieżki $P_u$ i~$P_w$ mają wspólny tylko wierzchołek $t$. Każda krawędź należy więc do~co najwyżej
jednej z~tych dwóch podścieżek. Krawędź $e$ jest incydentna z~$t$, zatem należy dokładnie do~jednej z~nich. Jeśli
$e\notin E(P_w)$, to z~Obserwacji~\ref{obs:shortest-paths}, zastosowanej do pary $t,w\in T$, wynika
\[
  E(P_w)\cap F\neq\emptyset.
\]
Ponieważ $e\notin E(P_w)$, każda krawędź z~$E(P_w)\cap F$ należy w~rzeczywistości do~$F^-$. Otrzymujemy
\[
  E(P)\cap F^-\neq\emptyset.
\]
Analogicznie, jeśli $e\notin E(P_u)$, to z~Obserwacji~\ref{obs:shortest-paths} dla pary $u,t\in T$ dostajemy
\[
  E(P_u)\cap F^-\neq\emptyset,
\]
a więc ponownie $E(P)\cap F^-\neq\emptyset$.

Wykazaliśmy, że każda ścieżka najkrótsza między różnymi wierzchołkami z~$T^-$ przecina zbiór $F^-$. Ponadto dla takich wierzchołków
odległość w~$G$ jest skończona, ponieważ były one już terminalami w~zbiorze $T$. Z~Obserwacji~\ref{obs:shortest-paths} wynika zatem,
że $F^-$ jest zbiorem semi-rozcinającym dla~$T^-$ w~$G$.

Para odpowiadająca trójce $(G,T^-,F^-)$ jest równa $(k-1,n)$, a więc jest mniejsza leksykograficznie od~$(k,n)$. Z~założenia
indukcyjnego otrzymujemy
\[
  |F^-|\ge |T^-|-1=(k-1)-1=k-2.
\]
Wobec tego
\[
  |F|=|F^-|+1\ge k-1.
\]
Krok indukcyjny jest zakończony, a tym samym prawdziwe jest zdanie $\mathcal{P}(k,n)$ dla każdej pary $(k,n)$.
\end{proof}

\begin{remark}
Oszacowanie z~Twierdzenia~\ref{thm:main} jest ostre. Jeżeli $G$ jest ścieżką oraz $T=V(G)$, to dla $F=E(G)$ zachodzi
$|F|=|V(G)|-1=|T|-1$, a~po usunięciu wszystkich krawędzi każda para różnych wierzchołków ma odległość równą $\infty$.
\end{remark}
