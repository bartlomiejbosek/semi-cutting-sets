\title{\textbf{Zbiory semi-rozcinające w grafach}}
\author{}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Udowadniamy, że dla dowolnego grafu $G=(V,E)$ oraz dowolnych zbiorów $T\subseteq V$ i~$F\subseteq E$, jeżeli usunięcie krawędzi ze zbioru
$F$ ściśle zwiększa odległość między każdą parą różnych wierzchołków z~$T$, to zachodzi nierówność $|F|\ge |T|-1$. Dowód opiera się na
charakteryzacji poprzez przecinanie najkrótszych ścieżek, własności zachowania tej własności przy ściąganiu domkniętego sąsiedztwa wierzchołka
nieincydentnego z~$F$, oraz na indukcji po parze $(|T|,|V|)$.
\end{abstract}

\section{Wstęp}

Niech $G=(V,E)$ będzie grafem, a~$T\subseteq V$ zbiorem \emph{terminali}. Interesuje nas, jak mały może być zbiór krawędzi
$F\subseteq E$ o~własności, że po jego usunięciu odległość między każdą parą różnych wierzchołków z~$T$ ulega ścisłemu zwiększeniu.
W~niniejszym tekście dowodzimy, że zawsze $|F|\ge |T|-1$. Kluczowym krokiem jest Propozycja~\ref{prop:contraction}, która pozwala
„ściskać” domknięte sąsiedztwo wierzchołka $t\in T$ nieincydentnego z~$F$ bez utraty własności semi-rozcinania; następnie stosujemy indukcję.

\section{Notacja i definicje}

W całym tekście $G=(V,E)$ oznacza skończony graf nieskierowany prosty: krawędzie utożsamiamy z~nieuporządkowanymi parami różnych
wierzchołków. Nie zakładamy spójności grafu.
Odległości traktujemy jako elementy zbioru $\mathbb{N}\cup\{\infty\}$ z~naturalnym porządkiem rozszerzonym o~$n<\infty$ dla
każdego $n\in\mathbb{N}$.

\begin{definition}[Ścieżki i odległość]\label{def:paths-distance}
Spacerem z~$u$ do~$w$ nazywamy ciąg wierzchołków $(v_0,v_1,\dots,v_k)$ taki, że $v_0=u$, $v_k=w$ oraz
$v_{i-1}v_i\in E$ dla $i=1,\dots,k$. Długość takiego spaceru wynosi $\len{(v_0,\dots,v_k)}\coloneqq k$.
Ścieżką nazywamy spacer bez powtórzeń wierzchołków.

Odległość $\dist{G}{u}{w}$ definiujemy jako najmniejszą długość spaceru z~$u$ do~$w$ w~$G$, o~ile taki spacer
istnieje; w~przeciwnym razie $\dist{G}{u}{w}\coloneqq\infty$.
\end{definition}

\begin{notation}[Wierzchołki i krawędzie ścieżki]\label{not:path-sets}
Jeżeli $P=(v_0,\dots,v_k)$ jest spacerem w~$G$, to oznaczamy
\[
  V(P)\coloneqq \{v_0,\dots,v_k\},
  \qquad
  E(P)\coloneqq \{v_{i-1}v_i:\ i=1,\dots,k\}.
\]
\end{notation}

\begin{notation}[Usuwanie krawędzi]\label{not:edge-removal}
Jeżeli $F\subseteq E$, to $G\setminus F$ oznacza graf $(V,E\setminus F)$.
\end{notation}

\begin{notation}[Końce krawędzi]\label{not:VF}
Dla $F\subseteq E$ przez $V[F]$ oznaczamy zbiór wierzchołków incydentnych z~krawędziami z~$F$, tj.
\[
  V[F]\coloneqq \{v\in V:\ \exists\, e\in F\ \text{takie, że $e$ jest incydentna z~$v$}\}.
\]
\end{notation}

\begin{notation}[Sąsiedztwo]\label{not:neighborhood}
Jeżeli $v\in V$, to $N(v)\coloneqq\{w\in V:\ vw\in E\}$ oraz $N[v]\coloneqq N(v)\cup\{v\}$.
\end{notation}

\begin{notation}[Ściąganie zbioru wierzchołków]\label{not:contraction}
Niech $W\subseteq V$ i~niech $w^\star\notin V$ będzie nowym wierzchołkiem. Definiujemy odwzorowanie
$\pi_W\colon V\to (V\setminus W)\cup\{w^\star\}$ dane wzorem
\[
  \pi_W(v)\coloneqq
  \begin{cases}
    w^\star & \text{gdy } v\in W,\\
    v & \text{gdy } v\notin W.
  \end{cases}
\]
Graf $G/W$ definiujemy jako $G/W\coloneqq (V',E')$, gdzie
\[
  V'\coloneqq (V\setminus W)\cup\{w^\star\},
  \qquad
  E'\coloneqq \{\pi_W(u)\pi_W(v):\ uv\in E,\ \pi_W(u)\neq \pi_W(v)\}.
\]
Krawędzie o obu końcach w~$W$ znikają (po ściągnięciu dawałyby pętle). Dla krawędzi $uv\in E$ z~co najwyżej
jednym końcem w~$W$ jej obraz w~$G/W$ oznaczamy przez $uv/W\coloneqq \pi_W(u)\pi_W(v)\in E'$.
Dla zbioru krawędzi $F\subseteq E$ definiujemy jego obraz
\[
  F/W\coloneqq \{\, \pi_W(u)\pi_W(v):\ uv\in F,\ \pi_W(u)\neq \pi_W(v)\,\}\subseteq E'.
\]
Ponieważ $E'$ i~$F/W$ są zbiorami, jeżeli różne krawędzie z~$F$ mają ten sam obraz w~$G/W$, to w~$F/W$ pojawia się on jednokrotnie.
\end{notation}

\begin{definition}[Zbiór semi-rozcinający]\label{def:semi-cut}
Niech $G=(V,E)$ będzie grafem. Mówimy, że $F\subseteq E$ jest zbiorem semi-rozcinającym dla $T\subseteq V$ w~$G$,
jeżeli dla każdej pary różnych wierzchołków $u,w\in T$ zachodzi
\[
  \dist{G}{u}{w} \;<\; \dist{G\setminus F}{u}{w}.
\]
\end{definition}

\section{Lematy i propozycje pomocnicze}

\begin{lemma}\label{lem:walk-to-path}
Jeżeli istnieje spacer z~$u$ do~$w$ w~$G$, to istnieje również ścieżka z~$u$ do~$w$ w~$G$ o~długości nie większej niż
długość tego spaceru. W~szczególności, jeżeli $\dist{G}{u}{w}<\infty$, to istnieje ścieżka z~$u$ do~$w$ o~długości
dokładnie $\dist{G}{u}{w}$.
\end{lemma}

\begin{proof}
Weźmy dowolny spacer $P=(v_0,\dots,v_k)$ z~$u$ do~$w$. Jeżeli w~$P$ występuje powtórzenie wierzchołka, to istnieją
indeksy $0\le i<j\le k$ takie, że $v_i=v_j$. Rozważmy ciąg
\[
  P'\coloneqq (v_0,\dots,v_i,v_{j+1},\dots,v_k),
\]
gdzie w przypadku $j=k$ ostatni fragment jest pusty. Jeżeli $j<k$, to ponieważ $v_i=v_j$ oraz $v_jv_{j+1}\in E$, mamy także
$v_iv_{j+1}\in E$. Zatem w~każdym przypadku $P'$ jest spacerem z~$u$ do~$w$ o~mniejszej długości niż $P$.
Powtarzając tę operację skończenie wiele razy, dostajemy spacer bez powtórzeń wierzchołków,
czyli ścieżkę, o~długości nie większej od $\len{P}$. Druga część wynika z~definicji odległości jako minimum po długościach
spacerów.
\end{proof}

\begin{lemma}\label{lem:subpath}
Niech $u,w\in V$ oraz niech $P=(v_0,\dots,v_k)$ będzie ścieżką z~$u$ do~$w$ w~$G$, tj.\ $v_0=u$ i~$v_k=w$. Załóżmy, że
$\len{P}=\dist{G}{u}{w}$. Niech $x=v_i$ i~$y=v_j$ dla pewnych $0\le i\le j\le k$ i~niech $P[x,y]\coloneqq (v_i,\dots,v_j)$ będzie
odpowiadającą podścieżką $P$ łączącą $x$ i~$y$. Wówczas $\len{P[x,y]}=\dist{G}{x}{y}$.
\end{lemma}

\begin{proof}
Niech $\len{P[x,y]}=L$. Gdyby istniała ścieżka $Q$ z~$x$ do~$y$ o~długości mniejszej niż $L$, to podmieniając w~$P$ odcinek
$P[x,y]$ na $Q$ otrzymalibyśmy spacer z~$u$ do~$w$ o~długości mniejszej niż $\len{P}=\dist{G}{u}{w}$, co jest sprzeczne z~definicją
odległości.
\end{proof}

\begin{observation}\label{obs:shortest-paths}
Niech $G=(V,E)$ będzie grafem oraz niech $F\subseteq E$ i~$T\subseteq V$. Wówczas $F$ jest zbiorem semi-rozcinającym dla~$T$
w~$G$ wtedy i~tylko wtedy, gdy dla każdych różnych $u,w\in T$ spełnione są łącznie następujące warunki:
$\dist{G}{u}{w}<\infty$ oraz każda ścieżka $P$ z~$u$ do~$w$ o~długości $\len{P}=\dist{G}{u}{w}$ zawiera co najmniej jedną
krawędź ze zbioru $F$.
\end{observation}

\begin{proof}
Ustalmy różne $u,w\in T$.

(\,$\Rightarrow$\,) Z~Definicji~\ref{def:semi-cut} mamy $\dist{G}{u}{w}<\dist{G\setminus F}{u}{w}\le\infty$, a~zatem
$\dist{G}{u}{w}<\infty$. Z~Lematu~\ref{lem:walk-to-path} wynika istnienie ścieżki z~$u$ do~$w$ o~długości $\dist{G}{u}{w}$; niech $P$
będzie dowolną taką ścieżką.
Gdyby $P$ nie zawierała żadnej krawędzi z~$F$, to byłaby również ścieżką w~$G\setminus F$ o~tej samej długości, co dawałoby
$\dist{G\setminus F}{u}{w}\le \dist{G}{u}{w}$, sprzecznie z~Definicją~\ref{def:semi-cut}.

(\,$\Leftarrow$\,) Załóżmy, że $\dist{G}{u}{w}<\infty$ oraz każda ścieżka $P$ z~$u$ do~$w$ o~długości $\dist{G}{u}{w}$
zawiera krawędź z~$F$. Jeżeli w~$G\setminus F$ nie istnieje żadna ścieżka z~$u$ do~$w$, to $\dist{G\setminus F}{u}{w}=\infty$ i~
nierówność $\dist{G}{u}{w}<\dist{G\setminus F}{u}{w}$ jest spełniona. W przeciwnym razie mamy $\dist{G\setminus F}{u}{w}<\infty$ i~
z~Lematu~\ref{lem:walk-to-path} wybierzmy ścieżkę $Q$ z~$u$ do~$w$ w~$G\setminus F$ o~długości $\len{Q}=\dist{G\setminus F}{u}{w}$.
Jest ona jednocześnie ścieżką w~$G$, a~zatem $\len{Q}\ge \dist{G}{u}{w}$. Ponadto z~definicji nie zawiera krawędzi z~$F$, więc nie może
mieć długości $\dist{G}{u}{w}$. Otrzymujemy $\len{Q}>\dist{G}{u}{w}$, a~ponieważ $\dist{G\setminus F}{u}{w}=\len{Q}$, dostajemy
$\dist{G\setminus F}{u}{w}>\dist{G}{u}{w}$. Ponieważ $u,w$ były dowolne, $F$ spełnia Definicję~\ref{def:semi-cut}.
\end{proof}

\begin{lemma}\label{lem:projection-walk}
Niech $G=(V,E)$ będzie grafem, $W\subseteq V$, a~$G/W$ zdefiniowany jak w~Notacji~\ref{not:contraction}.
Jeżeli $(v_0,\dots,v_k)$ jest spacerem w~$G$, to istnieje spacer $(w_0,\dots,w_m)$ w~$G/W$ taki, że
$w_0=\pi_W(v_0)$, $w_m=\pi_W(v_k)$ oraz $m\le k$.
\end{lemma}

\begin{proof}
Niech $a_i\coloneqq \pi_W(v_i)$ dla $i=0,\dots,k$. Zdefiniujemy indeksy $0=i_0<i_1<\dots<i_m\le k$ tak, aby
$(a_{i_0},a_{i_1},\dots,a_{i_m})$ było ciągiem otrzymanym z~$(a_0,\dots,a_k)$ przez usunięcie kolejnych powtórzeń.
Kładziemy $i_0\coloneqq 0$ oraz, jeżeli $i_r$ jest już zdefiniowany, a~$a_{i_r}\neq a_k$, to definiujemy
\[
  i_{r+1}\coloneqq \min\{j\in\{i_r+1,\dots,k\}:\ a_j\neq a_{i_r}\}.
\]
Konstrukcja kończy się, gdy $a_{i_m}=a_k$. Wtedy $w_r\coloneqq a_{i_r}$ dla $r=0,\dots,m$.
Z definicji mamy $w_0=a_0=\pi_W(v_0)$ oraz $w_m=a_k=\pi_W(v_k)$, a~ponieważ $i_0<i_1<\dots<i_m$, otrzymujemy $m\le k$.

Pozostaje sprawdzić, że $(w_0,\dots,w_m)$ jest spacerem w~$G/W$. Weźmy $r\in\{1,\dots,m\}$. Z definicji $i_r$ mamy
$a_{i_r}\neq a_{i_{r-1}}$ oraz $a_{i_r-1}=a_{i_{r-1}}$. Ponieważ $v_{i_r-1}v_{i_r}\in E$ i~$\pi_W(v_{i_r-1})\neq \pi_W(v_{i_r})$,
z definicji $E'$ wynika $w_{r-1}w_r=a_{i_{r-1}}a_{i_r}\in E'$. Zatem kolejne wierzchołki $w_{r-1},w_r$ są połączone krawędzią w~$G/W$,
więc $(w_0,\dots,w_m)$ jest spacerem.
\end{proof}

\begin{proposition}\label{prop:contraction}
Niech $G=(V,E)$ będzie grafem, $T\subseteq V$, $F\subseteq E$ i~niech $F$ będzie zbiorem semi-rozcinającym dla~$T$ w~$G$.
Niech $t\in T\setminus V[F]$ oraz niech $t'\notin V$ będzie nowym wierzchołkiem. Rozważmy graf $G'\coloneqq G/N[t]$, gdzie
$N[t]$ zostało ściągnięte do wierzchołka~$t'$, zbiór wierzchołków $T'\coloneqq (T\setminus\{t\})\cup\{t'\}$ oraz zbiór krawędzi
$F'\coloneqq F/N[t]\subseteq E(G')$. Wówczas $F'$ jest zbiorem semi-rozcinającym dla~$T'$ w~$G'$.
\end{proposition}

\begin{proof}
Najpierw pokażemy, że żaden wierzchołek z~$T\setminus\{t\}$ nie leży w~$N(t)$. Weźmy $u\in T\setminus\{t\}$.
Jeżeli $u\in N(t)$, to $\dist{G}{t}{u}=1$, a~każda ścieżka z~$t$ do~$u$ o~długości $1$ składa się z jednej krawędzi $tu$.
Z~Obserwacji~\ref{obs:shortest-paths} wynikałoby $tu\in F$, czyli $t\in V[F]$, sprzecznie z~założeniem.
Zatem $T\setminus\{t\}\subseteq V\setminus N[t]$ i~w szczególności $T'\subseteq V(G')$.

Pokażemy najpierw, że dla każdych różnych $u,w\in T'$ zachodzi $\dist{G'}{u}{w}<\infty$. Jeżeli $u,w\in T\setminus\{t\}$, to z
Definicji~\ref{def:semi-cut} otrzymujemy $\dist{G}{u}{w}<\infty$, a~więc istnieje spacer $Q$ z~$u$ do~$w$ w~$G$.
Z~Lematu~\ref{lem:projection-walk} wynika, że w~$G'$ istnieje spacer z~$u$ do~$w$, a~z Lematu~\ref{lem:walk-to-path} — że istnieje
również ścieżka z~$u$ do~$w$ w~$G'$. Jeżeli natomiast jeden z~wierzchołków $u,w$ jest równy $t'$, to drugi należy do $T\setminus\{t\}$;
oznaczmy go przez $z$. Z~Definicji~\ref{def:semi-cut} otrzymujemy $\dist{G}{t}{z}<\infty$, a~więc istnieje spacer z~$t$ do~$z$ w~$G$.
Stosując Lemat~\ref{lem:projection-walk} do tego spaceru dla $W=N[t]$ dostajemy spacer z~$t'$ do~$z$ w~$G'$, a~następnie z
Lematu~\ref{lem:walk-to-path} — ścieżkę z~$t'$ do~$z$ w~$G'$. W szczególności $\dist{G'}{t'}{z}<\infty$.

Ustalmy teraz różne $u,w\in T'$. Ponieważ $\dist{G'}{u}{w}<\infty$, z~Lematu~\ref{lem:walk-to-path} wynika istnienie ścieżki z~$u$ do~$w$
w~$G'$ o~długości $\dist{G'}{u}{w}$; niech $P'$ będzie dowolną taką ścieżką.
Pokażemy, że $E(P')\cap F'\neq\emptyset$.

\medskip\noindent
\textbf{Przypadek 1: $t'\notin V(P')$.}
Wtedy $V(P')\subseteq V\setminus N[t]$, a~z definicji $E(G')$ wynika $E(P')\subseteq E$. Zatem $P'$ jest również ścieżką w~$G$.
Gdyby w~$G$ istniał spacer $Q$ z~$u$ do~$w$ o~długości mniejszej niż $\len{P'}$,
to na mocy Lematu~\ref{lem:projection-walk} istniałby w~$G'$ spacer z~$u$ do~$w$ o~długości nie większej niż $\len{Q}<\len{P'}$,
co przeczyłoby wyborowi $P'$. Ponieważ $P'$ jest również spacerem w~$G$, otrzymujemy $\len{P'}=\dist{G}{u}{w}$.
Ponieważ $u,w\in T\setminus\{t\}$, z~Obserwacji~\ref{obs:shortest-paths} wynika, że $P'$ zawiera pewną krawędź $e\in F$.
Krawędź $e$ ma oba końce poza $N[t]$, więc $e/N[t]=e$ i~w szczególności $e\in F'$. Stąd $E(P')\cap F'\neq\emptyset$.

\medskip\noindent
\textbf{Przypadek 2: $t'\in V(P')$.}
Rozważmy najpierw podprzypadek, w~którym $t'$ jest jednym z~końców ścieżki $P'$. Niech $z$ będzie drugim końcem, tj.\
$\{z\}=\{u,w\}\setminus\{t'\}$. Jeżeli $P'$ nie zaczyna się w~$t'$, to odwróćmy kolejność wierzchołków na $P'$; otrzymujemy ścieżkę tej
samej długości i~o~tym samym zbiorze krawędzi, łączącą $t'$ z~$z$. Nadal oznaczamy ją przez $P'$. Zatem możemy zapisać
$P'$ w postaci $(t',x_1,\dots,x_k)$, gdzie $x_k=z$.
Z~definicji ściągania wynika, że krawędź $t'x_1$ w~$G'$ pochodzi od pewnej krawędzi $vx_1$ w~$G$ z~wierzchołkiem $v\in N[t]$.
Ponieważ $x_1\notin N[t]$, nie może zachodzić $v=t$ (wtedy $tx_1\in E$, więc $x_1\in N(t)\subseteq N[t]$). Zatem $v\in N(t)$, a~więc
$tv\in E(G)$.
Rozważmy ciąg wierzchołków $(t,v,x_1,\dots,x_k)$. Jest to ścieżka $P$ w~$G$: wierzchołek $v$ jest sąsiadem $t$, natomiast
$x_1,\dots,x_k$ nie należą do $N[t]$, więc są różne od $t$ i~$v$, a~ponieważ $P'$ jest ścieżką, wierzchołki $x_1,\dots,x_k$ są parami
różne. Mamy $\len{P}=\len{P'}+1$.

Niech $Q=(q_0,\dots,q_\ell)$ będzie dowolną ścieżką z~$t$ do~$z$ w~$G$, tj.\ $q_0=t$ oraz $q_\ell=z$. Ponieważ $z\notin N[t]$, mamy
$\ell\ge 2$, a~w szczególności $q_1\in N(t)$. Stąd $\pi_{N[t]}(q_0)=\pi_{N[t]}(q_1)=t'$. W dowodzie Lematu~\ref{lem:projection-walk}
konstruujemy indeksy $0=i_0<i_1<\dots<i_m=\ell$ odpowiadające usuwaniu kolejnych powtórzeń w~ciągu
$(\pi_{N[t]}(q_0),\dots,\pi_{N[t]}(q_\ell))$. Z równości $\pi_{N[t]}(q_0)=\pi_{N[t]}(q_1)$ wynika $i_1\ge 2$, a~zatem $m\le \ell-1$.
Otrzymujemy więc spacer z~$t'$ do~$z$ w~$G'$ o~długości nie większej niż $\ell-1=\len{Q}-1$. Stąd
$\dist{G'}{t'}{z}\le \len{Q}-1$, czyli $\len{Q}\ge \dist{G'}{t'}{z}+1=\len{P'}+1=\len{P}$. Otrzymujemy $\dist{G}{t}{z}=\len{P}$, a~zatem
$P$ jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.

Z~Obserwacji~\ref{obs:shortest-paths} ścieżka $P$ zawiera pewną krawędź $e\in F$. Pierwsza krawędź $tv$ ścieżki $P$ jest
incydentna z~$t$, więc nie należy do $F$ (bo $t\notin V[F]$). Zatem $e$ leży na odcinku $(v,x_1,\dots,x_k)$, który po ściągnięciu
$N[t]$ przechodzi w~odcinek $(t',x_1,\dots,x_k)$ ścieżki $P'$. Każda krawędź na tym odcinku ma co najwyżej jeden koniec w~$N[t]$, więc jej
obraz w~$G'$ jest zdefiniowany. W szczególności $e/N[t]$ jest krawędzią ścieżki $P'$ i~należy do $F'$, więc $E(P')\cap F'\neq\emptyset$.

Jeżeli natomiast $u\neq t'$, $w\neq t'$ oraz $t'$ jest wierzchołkiem wewnętrznym ścieżki $P'$, to $P'$ dzieli się na konkatenację
$P'=P'_1\cdot P'_2$, gdzie $P'_1$ łączy $u$ z~$t'$, a~$P'_2$ łączy $t'$ z~$w$. Z~Lematu~\ref{lem:subpath} wynika, że obie podścieżki
są najkrótsze między swoimi końcami. Stosujemy powyższy argument do ścieżki $P'_1$: jeżeli $P'_1$ nie zaczyna się w~$t'$, odwróćmy
kolejność wierzchołków na $P'_1$. Otrzymujemy $E(P'_1)\cap F'\neq\emptyset$. Następnie stosujemy ten sam argument do $P'_2$ i dostajemy
$E(P'_2)\cap F'\neq\emptyset$. W konsekwencji $E(P')\cap F'\neq\emptyset$.
\medskip

W~każdym przypadku wykazaliśmy $E(P')\cap F'\neq\emptyset$. Z~Obserwacji~\ref{obs:shortest-paths} wynika więc, że $F'$ jest zbiorem
semi-rozcinającym dla~$T'$ w~$G'$.
\end{proof}

\section{Główne twierdzenie}

\begin{theorem}\label{thm:main}
Niech $G=(V,E)$ będzie grafem, $T\subseteq V$ oraz $F\subseteq E$. Jeżeli $F$ jest zbiorem semi-rozcinającym dla~$T$ w~$G$, to
\[
  |F|\ge |T|-1.
\]
\end{theorem}

\begin{proof}
Dowodzimy tezy przez indukcję po parze $(|T|,|V|)$ w~porządku leksykograficznym, tj.\ $(a,b)\prec (c,d)$ wtedy i~tylko wtedy, gdy
$a<c$ lub $(a=c\ \text{i}\ b<d)$.

\medskip\noindent
\textbf{Baza indukcji.} Jeżeli $|T|\le 1$, to w~$T$ nie ma par różnych wierzchołków, więc warunek z~Definicji~\ref{def:semi-cut} jest
spełniony dla dowolnego $F$. W~takim przypadku $|F|\ge 0\ge |T|-1$.

\medskip\noindent
\textbf{Krok indukcyjny.} Niech $|T|\ge 2$ i~załóżmy, że twierdzenie jest prawdziwe dla wszystkich trójek $(G_0,T_0,F_0)$ z~parą
$(|T_0|,|V(G_0)|)$ mniejszą leksykograficznie od $(|T|,|V|)$. Rozróżnimy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $T\cap V[F]=\emptyset$.}
Wybierzmy dowolny $t\in T$ oraz dowolny $u\in T\setminus\{t\}$. Z~Definicji~\ref{def:semi-cut} wynika
$\dist{G}{t}{u}<\dist{G\setminus F}{t}{u}\le\infty$, a~zatem $\dist{G}{t}{u}<\infty$. Istnieje więc spacer $(v_0,\dots,v_k)$ z~$v_0=t$
do~$v_k=u$. Ponieważ $u\neq t$, mamy $k\ge 1$, więc $v_1\in N(t)$ i~$|N[t]|\ge 2$.

Niech $G'\coloneqq G/N[t]$, $T'\coloneqq (T\setminus\{t\})\cup\{t'\}$ oraz $F'\coloneqq F/N[t]$ jak w~Propozycji~\ref{prop:contraction}.
Na mocy tej propozycji $F'$ jest zbiorem semi-rozcinającym dla~$T'$ w~$G'$. Ponadto $|T'|=|T|$ oraz, ponieważ $|N[t]|\ge 2$,
zachodzi $|V(G')|=|V|-|N[t]|+1<|V|$. Zatem z~założenia indukcyjnego otrzymujemy
\[
  |F'|\ge |T'|-1=|T|-1.
\]
Z~definicji obrazu zbioru krawędzi przy ściąganiu mamy $|F'|\le |F|$, a~więc $|F|\ge |T|-1$.

\medskip\noindent
\textbf{Przypadek 2: $T\cap V[F]\neq\emptyset$.}
Wybierzmy $t\in T\cap V[F]$ oraz krawędź $e\in F$ incydentną z~$t$. Zdefiniujmy $T^-\coloneqq T\setminus\{t\}$ oraz
$F^-\coloneqq F\setminus\{e\}$. Pokażemy, że $F^-$ jest zbiorem semi-rozcinającym dla~$T^-$ w~$G$.

Weźmy dowolne różne $u,w\in T^-$. Ponieważ $u,w\in T$, z~Definicji~\ref{def:semi-cut} otrzymujemy $\dist{G}{u}{w}<\infty$.
Z~Lematu~\ref{lem:walk-to-path} wynika istnienie ścieżki z~$u$ do~$w$ w~$G$ o~długości $\dist{G}{u}{w}$; niech $P$ będzie dowolną taką
ścieżką.
Z~Obserwacji~\ref{obs:shortest-paths} wynika, że $P$ zawiera pewną krawędź ze zbioru $F$.
Jeżeli $e\notin E(P)$, to mamy od razu $E(P)\cap F^-\neq\emptyset$. Załóżmy więc, że $e\in E(P)$. Wtedy $t$ leży na~$P$ i~ścieżka
$P$ rozkłada się na konkatenację $P=P_u\cdot P_w$, gdzie $P_u$ łączy $u$ z~$t$, a~$P_w$ łączy $t$ z~$w$. Z~Lematu~\ref{lem:subpath}
wynika, że $\len{P_u}=\dist{G}{u}{t}$ oraz $\len{P_w}=\dist{G}{t}{w}$. Ponieważ $P$ jest ścieżką, podścieżki $P_u$ i~$P_w$ mają wspólny
jedynie wierzchołek $t$, a~zatem $E(P_u)\cap E(P_w)=\emptyset$ oraz $E(P)=E(P_u)\,\dot\cup\,E(P_w)$. Krawędź $e$ jest incydentna z~$t$,
więc należy dokładnie do jednej z~podścieżek $P_u$ i~$P_w$. Jeżeli $e\notin E(P_w)$, to ponieważ $t,w\in T$, z~Obserwacji~\ref{obs:shortest-paths}
wynika $E(P_w)\cap F\neq\emptyset$, a~zatem $E(P_w)\cap F^-\neq\emptyset$, co implikuje $E(P)\cap F^-\neq\emptyset$. Jeżeli natomiast
$e\notin E(P_u)$, to ponieważ $u,t\in T$, dostajemy $E(P_u)\cap F^-\neq\emptyset$, więc $E(P)\cap F^-\neq\emptyset$.

Ponieważ $u,w$ były dowolne, z~Obserwacji~\ref{obs:shortest-paths} wynika, że $F^-$ jest zbiorem semi-rozcinającym dla~$T^-$ w~$G$.
Teraz para $(|T^-|,|V|)=(|T|-1,|V|)$ jest mniejsza leksykograficznie niż $(|T|,|V|)$, więc z~założenia indukcyjnego
\[
  |F^-|\ge |T^-|-1=|T|-2.
\]
W konsekwencji $|F|=|F^-|+1\ge |T|-1$, co kończy dowód.
\end{proof}

\begin{remark}
Nierówność z~Twierdzenia~\ref{thm:main} jest ostra. Dla ścieżki $G$ o~zbiorze wierzchołków $V$ i~$T=V$ możemy wziąć $F=E$; wówczas
$|F|=|V|-1=|T|-1$, a~po usunięciu $F$ odległości między różnymi wierzchołkami stają się równe $\infty$.
\end{remark}
