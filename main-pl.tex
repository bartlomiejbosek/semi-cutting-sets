\title{\textbf{O rozmiarze zbiorów na wpół-tnących w grafach}}
\author{}
\date{}

\begin{document}
\maketitle

\begin{abstract}
Niech $H=(V,E)$ będzie skończonym grafem nieskierowanym, niech $T\subseteq V(H)$ oraz $F\subseteq E(H)$. Zakładamy, że dla
każdych różnych wierzchołków $u,w\in T$ zachodzi nierówność
\[
  \dist{H}{u}{w}<\dist{H\setminus F}{u}{w}.
\]
Udowadniamy, że wówczas $|F|\ge |T|-1$. Dowód opiera się na charakteryzacji zbiorów na wpół-tnących przez przecinanie
wszystkich ścieżek najkrótszych, na niezmienniczości tej własności przy ściąganiu domkniętego sąsiedztwa terminala
nieincydentnego z~$F$ oraz na indukcji leksykograficznej względem pary $(|T|,|V(H)|)$.
\end{abstract}

\section{Wprowadzenie}

Niech $H=(V,E)$ będzie skończonym grafem nieskierowanym, $T\subseteq V(H)$ oraz $F\subseteq E(H)$. Mówimy, że $F$ jest
\emph{zbiorem na wpół-tnącym} dla $T$, jeżeli dla każdej pary różnych wierzchołków $u,w\in T$ po usunięciu krawędzi z~$F$
odległość między $u$ i~$w$ ściśle rośnie. Przedmiotem pracy jest formalny dowód twierdzenia
\[
  |F|\ge |T|-1,
\]
ważnego dla wszystkich grafów skończonych, bez założeń o spójności, planarności ani innej szczególnej strukturze.

Argument ma trzy składniki. Najpierw zapisujemy równoważną postać definicji przez przecinanie wszystkich ścieżek najkrótszych.
Następnie wykazujemy, że własność ta zachowuje się po ściągnięciu domkniętego sąsiedztwa terminala nieincydentnego z~$F$.
Na końcu przeprowadzamy indukcję leksykograficzną względem pary $(|T|,|V(H)|)$. Dla wygody czytelnika cały wywód, łącznie
z~lematami elementarnymi, zapisujemy w~sposób samodzielny.

W~Sekcji~2 ustalamy notację. Sekcja~3 zawiera lematy pomocnicze o~ścieżkach najkrótszych. W~Sekcji~4 dowodzimy
niezmienniczości własności pół-tnącej przy ściąganiu, a~w~Sekcji~5 przeprowadzamy dowód twierdzenia głównego.

\section{Notacja i definicje}

O~ile jawnie nie zaznaczono inaczej, przez graf rozumiemy skończony graf nieskierowany prosty, to znaczy graf bez pętli
i~bez krawędzi równoległych. W~takim grafie każdą krawędź utożsamiamy z~dwuelementowym podzbiorem zbioru wierzchołków.
Jeżeli $e\in E$ oraz $v\in e$, to mówimy, że $v$ jest końcem krawędzi $e$. Odległości traktujemy jako elementy zbioru
$\mathbb{N}_0\cup\{\infty\}$, gdzie $\mathbb{N}_0=\{0,1,2,\dots\}$, z~naturalnym porządkiem rozszerzonym o~relacje
$n<\infty$ dla każdego $n\in\mathbb{N}_0$.

\begin{definition}[Spacery, ścieżki i odległość]\label{def:paths-distance}
Niech $u,w\in V$. Spacerem z~$u$ do~$w$ nazywamy ciąg wierzchołków
\[
  P=(v_0,v_1,\dots,v_k),
\]
taki że $v_0=u$, $v_k=w$ oraz $\{v_{i-1},v_i\}\in E$ dla każdego $i\in\{1,\dots,k\}$. Długość spaceru $P$ definiujemy przez
$\len{P}\coloneqq k$.

Ścieżką nazywamy spacer $P=(v_0,\dots,v_k)$, dla którego wierzchołki $v_0,\dots,v_k$ są parami różne. Dla ścieżki
$P=(v_0,\dots,v_k)$ oznaczamy
\[
  V(P)\coloneqq \{v_0,\dots,v_k\},
  \qquad
  E(P)\coloneqq \bigl\{\{v_{i-1},v_i\}: i=1,\dots,k\bigr\}.
\]

Jeżeli $P=(v_0,\dots,v_k)$ oraz $Q=(w_0,\dots,w_\ell)$ są spacerami takimi, że $v_k=w_0$, to ich konkatenację definiujemy przez
\[
  P\cdot Q\coloneqq (v_0,\dots,v_k,w_1,\dots,w_\ell).
\]

Odległość $\dist{G}{u}{w}$ definiujemy jako minimum długości ścieżek z~$u$ do~$w$ w~$G$, jeżeli taka ścieżka istnieje; w przeciwnym
razie kładziemy $\dist{G}{u}{w}\coloneqq \infty$.
\end{definition}

\begin{notation}[Podstawowe operacje na grafie]\label{not:basic-operations}
Jeżeli $F\subseteq E$, to przez $G\setminus F$ oznaczamy graf $(V,E\setminus F)$. Dla $F\subseteq E$ piszemy
\[
  V[F]\coloneqq \{v\in V:\ \exists\, e\in F\ \text{ takie, że }\ v\in e\}.
\]
Jeżeli $v\in V$, to jego otoczenie i~domknięte otoczenie oznaczamy odpowiednio przez
\[
  N(v)\coloneqq \{w\in V:\ \{v,w\}\in E\},
  \qquad
  N[v]\coloneqq N(v)\cup\{v\}.
\]
Jeżeli $P=(v_0,\dots,v_k)$ jest ścieżką i~$0\le i\le j\le k$, to oznaczamy
\[
  P[v_i,v_j]\coloneqq (v_i,\dots,v_j).
\]
\end{notation}

\begin{definition}[Ściąganie zbioru wierzchołków]\label{def:contraction}
Niech $W\subseteq V$ oraz niech $w^\ast\notin V$ będzie nowym wierzchołkiem. Definiujemy odwzorowanie
\[
  \pi_W\colon V\to (V\setminus W)\cup\{w^\ast\},
  \qquad
  \pi_W(v)\coloneqq
  \begin{cases}
    w^\ast & \text{dla } v\in W,\\
    v & \text{dla } v\notin W.
  \end{cases}
\]
Przez $G/W$ oznaczamy graf $(V',E')$, gdzie
\[
  V'\coloneqq (V\setminus W)\cup\{w^\ast\},
  \qquad
  E'\coloneqq \bigl\{\{\pi_W(x),\pi_W(y)\}: \{x,y\}\in E,\ \pi_W(x)\neq \pi_W(y)\bigr\}.
\]
Jeżeli $e=\{x,y\}\in E$ oraz $\pi_W(x)\neq \pi_W(y)$, to obraz tej krawędzi oznaczamy przez
\[
  e/W\coloneqq \{\pi_W(x),\pi_W(y)\}\in E'.
\]
Dla $F\subseteq E$ definiujemy obraz zbioru krawędzi przez
\[
  F/W\coloneqq \bigl\{\{\pi_W(x),\pi_W(y)\}: \{x,y\}\in F,\ \pi_W(x)\neq \pi_W(y)\bigr\}\subseteq E'.
\]
\end{definition}

\begin{lemma}\label{lem:contraction-image-size}
Niech $W\subseteq V$ oraz $F\subseteq E$. Wówczas odwzorowanie
\[
  \varphi\colon \bigl\{\{x,y\}\in F:\ \pi_W(x)\neq \pi_W(y)\bigr\}\to F/W,
  \qquad
  \varphi(\{x,y\})\coloneqq \{\pi_W(x),\pi_W(y)\},
\]
jest surjekcją. W~szczególności
\[
  |F/W|\le |F|.
\]
\end{lemma}

\begin{proof}
Surjektywność wynika bezpośrednio z~definicji zbioru $F/W$: każdy jego element ma postać $\{\pi_W(x),\pi_W(y)\}$ dla pewnej
krawędzi $\{x,y\}\in F$ spełniającej $\pi_W(x)\neq \pi_W(y)$. Dziedzina odwzorowania $\varphi$ jest podzbiorem zbioru $F$, zatem
\[
  |F/W|\le \left|\bigl\{\{x,y\}\in F:\ \pi_W(x)\neq \pi_W(y)\bigr\}\right|\le |F|.
\]
\end{proof}

\begin{definition}[Zbiór na wpół-tnący]\label{def:semi-cut}
Niech $T\subseteq V$. Mówimy, że $F\subseteq E$ jest zbiorem na wpół-tnącym dla~$T$ w~$G$, jeżeli dla każdej pary różnych
wierzchołków $u,w\in T$ zachodzi
\[
  \dist{G}{u}{w}<\dist{G\setminus F}{u}{w}.
\]
\end{definition}

\begin{lemma}[Redukcja do grafów prostych]\label{lem:simple-reduction}
Niech $H$ będzie dowolnym skończonym grafem nieskierowanym, również z~pętlami i~krawędziami równoległymi, oraz niech
$T\subseteq V(H)$ i~$F\subseteq E(H)$. Oznaczmy przez $H^{\mathrm s}$ graf prosty otrzymany z~$H$ przez usunięcie wszystkich pętli
i~zastąpienie każdej niepustej klasy krawędzi równoległych pojedynczą krawędzią. Dla $F\subseteq E(H)$ definiujemy
$F^{\mathrm s}\subseteq E(H^{\mathrm s})$ jako zbiór tych krawędzi $uv\in E(H^{\mathrm s})$, dla których każda krawędź grafu $H$
o~końcach $u$ i~$v$ należy do~$F$. Wtedy dla każdych wierzchołków $u,w\in V(H)$ zachodzą równości
\[
  \dist{H}{u}{w}=\dist{H^{\mathrm s}}{u}{w}
  \qquad\text{oraz}\qquad
  \dist{H\setminus F}{u}{w}=\dist{H^{\mathrm s}\setminus F^{\mathrm s}}{u}{w}.
\]
W szczególności, jeżeli $F$ jest zbiorem na wpół-tnącym dla~$T$ w~$H$, to $F^{\mathrm s}$ jest zbiorem na wpół-tnącym dla~$T$
w~$H^{\mathrm s}$, a ponadto
\[
  |F^{\mathrm s}|\le |F|.
\]
\end{lemma}

\begin{proof}
Pętle nie należą do żadnej ścieżki między różnymi wierzchołkami. Gdy $u=w$, obie rozważane odległości są równe $0$, więc pozostaje
przypadek $u\neq w$. Każda ścieżka z~$u$ do~$w$ w~$H$ może używać co najwyżej jednej krawędzi z~każdej klasy krawędzi równoległych
o~tych samych końcach; po zastąpieniu każdej takiej krawędzi reprezentantem odpowiedniej klasy dostajemy ścieżkę tej samej
długości w~$H^{\mathrm s}$. Odwrotnie, każda ścieżka w~$H^{\mathrm s}$ podnosi się do ścieżki tej samej długości w~$H$ przez wybór
dowolnego reprezentanta każdej użytej klasy. Stąd
\[
  \dist{H}{u}{w}=\dist{H^{\mathrm s}}{u}{w}.
\]

Po usunięciu zbioru $F$ sytuacja jest analogiczna. Dla ustalonych końców $u$ i~$v$ krawędź $uv$ należy do
$E(H^{\mathrm s}\setminus F^{\mathrm s})$ wtedy i~tylko wtedy, gdy w~$H\setminus F$ pozostaje przynajmniej jedna krawędź o~końcach
$u$ i~$v$. Rzeczywiście, jeżeli $uv\notin F^{\mathrm s}$, to z~definicji $F^{\mathrm s}$ wynika istnienie krawędzi o~końcach $u$ i~$v$,
która nie należy do~$F$; odwrotnie, jeżeli taka krawędź istnieje w~$H\setminus F$, to nie wszystkie krawędzie o~końcach $u$ i~$v$
leżą w~$F$, więc $uv\notin F^{\mathrm s}$. Zatem ścieżki w~$H\setminus F$ i~$H^{\mathrm s}\setminus F^{\mathrm s}$ również
odpowiadają sobie z zachowaniem długości, skąd
\[
  \dist{H\setminus F}{u}{w}=\dist{H^{\mathrm s}\setminus F^{\mathrm s}}{u}{w}.
\]
Te dwie równości odległości dają natychmiast wniosek o~zachowaniu własności pół-tnącej. Dla oszacowania liczności zauważmy, że
jeżeli $uv\in F^{\mathrm s}$, to z~definicji $F^{\mathrm s}$ każda krawędź grafu $H$ o~końcach $u$ i~$v$ należy do~$F$. Wybierzmy
jedną z~nich i~oznaczmy ją przez $e_{uv}$. Różne krawędzie grafu $H^{\mathrm s}$ mają różne pary końców, więc odwzorowanie
$uv\mapsto e_{uv}$ jest injekcją $F^{\mathrm s}\to F$. Otrzymujemy $|F^{\mathrm s}|\le |F|$.
\end{proof}

\section{Lematy pomocnicze}

\begin{lemma}\label{lem:walk-to-path}
Niech $H$ będzie grafem prostym. Jeżeli w~$H$ istnieje spacer z~$u$ do~$w$, to istnieje również ścieżka z~$u$ do~$w$ w~$H$
o~długości nie większej niż długość tego spaceru.
\end{lemma}

\begin{proof}
Niech $Q$ będzie ustalonym spacerem z~$u$ do~$w$ w~$H$. Wśród wszystkich spacerów z~$u$ do~$w$ o~długości nie większej niż $\len{Q}$
wybierzmy spacer $P=(v_0,\dots,v_k)$ o~najmniejszej długości. Pokażemy, że $P$ jest ścieżką.

Przypuśćmy przeciwnie, że istnieją indeksy $0\le i<j\le k$ takie, że $v_i=v_j$. Jeżeli $j<k$, to z~$\{v_j,v_{j+1}\}\in E$ oraz
$v_i=v_j$ wynika $\{v_i,v_{j+1}\}\in E$, więc ciąg
\[
  P'=(v_0,\dots,v_i,v_{j+1},\dots,v_k)
\]
jest spacerem z~$u$ do~$w$ o~długości mniejszej niż $\len{P}$. Jeżeli natomiast $j=k$, to ciąg
\[
  P''=(v_0,\dots,v_i)
\]
jest spacerem z~$u$ do~$w$, ponieważ $v_i=v_k=w$, i~również ma długość mniejszą niż $\len{P}$. W~obu sytuacjach otrzymujemy
sprzeczność z~minimalnością wyboru spaceru $P$.
Zatem $P$ nie ma powtórzeń wierzchołków, czyli jest ścieżką.
\end{proof}

\begin{lemma}\label{lem:subpath}
Niech $H$ będzie grafem prostym oraz niech $P=(v_0,\dots,v_k)$ będzie ścieżką z~$u$ do~$w$ w~$H$ o~długości
$\len{P}=\dist{H}{u}{w}$. Wówczas dla każdych indeksów $0\le i\le j\le k$ zachodzi
\[
  \len{P[v_i,v_j]}=\dist{H}{v_i}{v_j}.
\]
\end{lemma}

\begin{proof}
Załóżmy przeciwnie, że istnieje ścieżka $Q$ z~$v_i$ do~$v_j$ taka, że
\[
  \len{Q}<\len{P[v_i,v_j]}.
\]
Wtedy ciąg
\[
  (v_0,\dots,v_i)\cdot Q\cdot (v_j,\dots,v_k)
\]
jest spacerem z~$u$ do~$w$ w~$H$ o~długości
\[
  i+\len{Q}+(k-j)
  <
  i+\len{P[v_i,v_j]}+(k-j)
  =
  \len{P}
  =
  \dist{H}{u}{w}.
\]
Na mocy Lematu~\ref{lem:walk-to-path} istnieje więc w~$H$ ścieżka z~$u$ do~$w$ o~długości mniejszej niż $\dist{H}{u}{w}$, co jest
sprzeczne z~definicją odległości. Zatem podścieżka $P[v_i,v_j]$ jest ścieżką najkrótszą między swoimi końcami.
\end{proof}

\begin{observation}\label{obs:shortest-paths}
Niech $T\subseteq V$ oraz $F\subseteq E$. Wówczas $F$ jest zbiorem na wpół-tnącym dla~$T$ w~$G$ wtedy i~tylko wtedy, gdy dla
każdych różnych $u,w\in T$ zachodzi $\dist{G}{u}{w}<\infty$ oraz każda ścieżka $P$ z~$u$ do~$w$ o~długości
$\len{P}=\dist{G}{u}{w}$ spełnia $E(P)\cap F\neq\emptyset$.
\end{observation}

\begin{proof}
Ustalmy różne wierzchołki $u,w\in T$.

Załóżmy najpierw, że $F$ jest zbiorem na wpół-tnącym dla~$T$. Wtedy
\[
  \dist{G}{u}{w}<\dist{G\setminus F}{u}{w}\le \infty,
\]
więc $\dist{G}{u}{w}<\infty$. Niech $P$ będzie dowolną ścieżką z~$u$ do~$w$ o~długości $\dist{G}{u}{w}$. Gdyby
$E(P)\cap F=\emptyset$, to $P$ byłaby także ścieżką w~$G\setminus F$, a~zatem
\[
  \dist{G\setminus F}{u}{w}\le \len{P}=\dist{G}{u}{w},
\]
co przeczy założeniu.

Załóżmy teraz, że $\dist{G}{u}{w}<\infty$ oraz każda ścieżka z~$u$ do~$w$ o~długości $\dist{G}{u}{w}$ przecina zbiór $F$.
Jeżeli w~$G\setminus F$ nie istnieje ścieżka z~$u$ do~$w$, to
$\dist{G\setminus F}{u}{w}=\infty$, a~więc
\[
  \dist{G}{u}{w}<\infty=\dist{G\setminus F}{u}{w}.
\]
Pozostaje przypadek $\dist{G\setminus F}{u}{w}<\infty$.
Niech $Q$ będzie ścieżką z~$u$ do~$w$ w~$G\setminus F$ o~długości
\[
  \len{Q}=\dist{G\setminus F}{u}{w}.
\]
Ponieważ $Q$ jest także ścieżką w~$G$, mamy $\len{Q}\ge \dist{G}{u}{w}$. Równość nie może zachodzić, gdyż wtedy $Q$ byłaby ścieżką
najkrótszą w~$G$ nieprzecinającą zbioru $F$, wbrew przyjętemu założeniu. Dostajemy więc
\[
  \dist{G\setminus F}{u}{w}=\len{Q}>\dist{G}{u}{w}.
\]
Ponieważ para $u,w$ była dowolna, zbiór $F$ jest zbiorem na wpół-tnącym dla~$T$.
\end{proof}

\begin{lemma}\label{lem:projection-walk}
Niech $W\subseteq V$. Jeżeli $P=(v_0,\dots,v_k)$ jest spacerem w~$G$, to w~$G/W$ istnieje spacer
\[
  P'=(w_0,\dots,w_m)
\]
taki, że $w_0=\pi_W(v_0)$, $w_m=\pi_W(v_k)$ oraz $m\le k$.
\end{lemma}

\begin{proof}
Niech $a_i\coloneqq \pi_W(v_i)$ dla $i=0,\dots,k$. Niech
\[
  0=i_0<i_1<\dots<i_m\le k
\]
będą pierwszymi indeksami kolejnych maksymalnych bloków stałych w~ciągu $(a_0,\dots,a_k)$. Innymi słowy, dla każdego
$r\in\{0,\dots,m-1\}$ zachodzi
\[
  a_{i_r}=a_{i_r+1}=\dots=a_{i_{r+1}-1}
  \qquad\text{oraz}\qquad
  a_{i_r}\neq a_{i_{r+1}},
\]
a ponadto $a_{i_m}=a_k$. Definiujemy
\[
  w_r\coloneqq a_{i_r}
  \qquad\text{dla } r=0,\dots,m.
\]
Wtedy $w_0=a_0=\pi_W(v_0)$, $w_m=a_k=\pi_W(v_k)$, a~ponieważ
\[
  0=i_0<i_1<\dots<i_m\le k,
\]
otrzymujemy także $m\le k$.

Pozostaje sprawdzić, że kolejne wierzchołki $w_r,w_{r+1}$ są połączone krawędzią w~$G/W$. Dla
$r\in\{0,\dots,m-1\}$ mamy
\[
  a_{i_{r+1}-1}=a_{i_r}
  \qquad\text{oraz}\qquad
  a_{i_{r+1}-1}\neq a_{i_{r+1}}.
\]
Ponieważ $\{v_{i_{r+1}-1},v_{i_{r+1}}\}\in E$, z~Definicji~\ref{def:contraction} wynika, że
\[
  \{w_r,w_{r+1}\}
  =
  \{\pi_W(v_{i_{r+1}-1}),\pi_W(v_{i_{r+1}})\}
  \in E(G/W).
\]
Zatem $P'$ jest spacerem w~$G/W$.
\end{proof}

\section{Zachowanie własności przy ściąganiu}

\begin{lemma}\label{lem:lifting-endpoint}
Niech $t\in V$, niech $t^\prime\notin V$ będzie nowym wierzchołkiem oraz niech $G^\prime\coloneqq G/N[t]$, gdzie zbiór $N[t]$ został
ściągnięty do~$t^\prime$. Niech $z\in V\setminus N[t]$ i~niech
\[
  P^\prime=(t^\prime,x_1,\dots,x_k)
\]
będzie ścieżką najkrótszą z~$t^\prime$ do~$z$ w~$G^\prime$. Wówczas istnieje wierzchołek $v\in N(t)$ taki, że ciąg
\[
  P=(t,v,x_1,\dots,x_k)
\]
jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.
\end{lemma}

\begin{proof}
Ponieważ $z\neq t^\prime$, mamy $k\ge 1$. Wierzchołek $x_1$ należy do~$V(G^\prime)\setminus\{t^\prime\}$, zatem
$x_1\in V\setminus N[t]$. Krawędź $\{t^\prime,x_1\}$ należy do~$E(G^\prime)$, więc z~Definicji~\ref{def:contraction}
wynika istnienie krawędzi $\{v,x_1\}\in E(G)$ takiej, że $v\in N[t]$ oraz $\{v,x_1\}/N[t]=\{t^\prime,x_1\}$. Gdyby $v=t$, to
z~$\{v,x_1\}=\{t,x_1\}\in E(G)$ otrzymalibyśmy $x_1\in N(t)\subseteq N[t]$, sprzecznie z~$x_1\in V\setminus N[t]$. Zatem
$v\neq t$, czyli $v\in N(t)$.

Pokażemy, że
\[
  P=(t,v,x_1,\dots,x_k)
\]
jest ścieżką w~$G$. Wierzchołek $v$ jest sąsiadem $t$, więc $\{t,v\}\in E$. Dla każdego $i\in\{1,\dots,k\}$ wierzchołek
$x_i$ należy do~$V\setminus N[t]$, a~więc $x_i\neq t$ oraz $x_i\neq v$. Ponieważ $P^\prime$ jest ścieżką, wierzchołki
$x_1,\dots,x_k$ są parami różne. Ponadto dla każdego $i\in\{2,\dots,k\}$ krawędź $\{x_{i-1},x_i\}$ należy do~$E(G^\prime)$
i~ma oba końce poza $N[t]$, więc z~Definicji~\ref{def:contraction} wynika $\{x_{i-1},x_i\}\in E(G)$. Wobec tego $P$ jest ścieżką
i~spełnia
\[
  \len{P}=\len{P^\prime}+1.
\]

Niech teraz $Q=(q_0,\dots,q_\ell)$ będzie dowolną ścieżką z~$t$ do~$z$ w~$G$. Ponieważ $z\notin N[t]$ oraz $q_0=t$, mamy $\ell\ge 2$
i~$q_1\in N(t)$. Stąd $\pi_{N[t]}(q_1)=t^\prime$. Zastosowanie Lematu~\ref{lem:projection-walk} do spaceru
\[
  (q_1,\dots,q_\ell)
\]
daje w~$G^\prime$ spacer z~$t^\prime$ do~$z$ o~długości co najwyżej $\ell-1$. Z definicji odległości dostajemy
\[
  \dist{G^\prime}{t^\prime}{z}\le \ell-1=\len{Q}-1.
\]
Ponieważ ostatnia nierówność zachodzi dla każdej ścieżki $Q$ z~$t$ do~$z$ w~$G$, po wzięciu minimum po wszystkich takich ścieżkach
otrzymujemy
\[
  \dist{G}{t}{z}\ge \dist{G^\prime}{t^\prime}{z}+1.
\]
Z drugiej strony $P$ jest ścieżką z~$t$ do~$z$, więc
\[
  \dist{G}{t}{z}\le \len{P}=\len{P^\prime}+1=\dist{G^\prime}{t^\prime}{z}+1.
\]
Obie nierówności dają równość
\[
  \dist{G}{t}{z}=\len{P},
\]
czyli $P$ jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.
\end{proof}

\begin{proposition}\label{prop:contraction}
Niech $T\subseteq V$ oraz $F\subseteq E$. Załóżmy, że $F$ jest zbiorem na wpół-tnącym dla~$T$ w~$G$. Niech
$t\in T\setminus V[F]$ oraz niech $t^\prime\notin V$ będzie nowym wierzchołkiem. Oznaczmy
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
Wówczas $F^\prime$ jest zbiorem na wpół-tnącym dla~$T^\prime$ w~$G^\prime$.
\end{proposition}

\begin{proof}
Najpierw pokażemy, że
\[
  T\setminus\{t\}\subseteq V\setminus N[t].
\]
Weźmy dowolne $u\in T\setminus\{t\}$ i~przypuśćmy, że $u\in N(t)$. Wtedy $\dist{G}{t}{u}=1$, a~jedyna ścieżka długości $1$ z~$t$
do~$u$ jest równa $(t,u)$, ponieważ w~grafie prostym istnieje dokładnie jedna krawędź o~końcach $t$ i~$u$. Z~Obserwacji
\ref{obs:shortest-paths} wynikałoby, że krawędź $\{t,u\}$ należy do~$F$, co przeczy założeniu $t\notin V[F]$. Otrzymana
sprzeczność dowodzi, że żaden terminal różny od~$t$ nie należy do~$N[t]$. W~szczególności
$T^\prime\subseteq V(G^\prime)$.

Pokażemy teraz, że dla każdych różnych $u,w\in T^\prime$ zachodzi $\dist{G^\prime}{u}{w}<\infty$. Jeżeli
$u,w\in T\setminus\{t\}$, to z~Definicji~\ref{def:semi-cut} wynika $\dist{G}{u}{w}<\infty$, więc w~$G$ istnieje ścieżka
z~$u$ do~$w$. Z~Lematu~\ref{lem:projection-walk} otrzymujemy spacer z~$u$ do~$w$ w~$G^\prime$, a następnie z~Lematu~\ref{lem:walk-to-path}
ścieżkę z~$u$ do~$w$ w~$G^\prime$. Jeżeli jeden z~wierzchołków $u,w$ jest równy $t^\prime$, to drugi należy do~$T\setminus\{t\}$;
oznaczmy go przez $z$. Wtedy z~Definicji~\ref{def:semi-cut} mamy $\dist{G}{t}{z}<\infty$, więc w~$G$ istnieje ścieżka z~$t$ do~$z$.
Jej obraz daje, po użyciu lematów \ref{lem:projection-walk} i~\ref{lem:walk-to-path}, ścieżkę z~$t^\prime$ do~$z$ w~$G^\prime$.

Niech teraz $u,w\in T^\prime$ będą różne i~niech $P^\prime$ będzie dowolną ścieżką najkrótszą z~$u$ do~$w$ w~$G^\prime$. Zgodnie
z~Obserwacją~\ref{obs:shortest-paths} wystarczy wykazać, że
\[
  E(P^\prime)\cap F^\prime\neq\emptyset.
\]
Rozpatrzymy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $t^\prime\notin V(P^\prime)$.}
Ponieważ
\[
  V(G^\prime)=(V\setminus N[t])\cup\{t^\prime\},
\]
z~założenia $t^\prime\notin V(P^\prime)$ wynika
\[
  V(P^\prime)\subseteq V\setminus N[t].
\]
Jeżeli $f=\{x,y\}\in E(P^\prime)$, to $x,y\notin N[t]$. Mamy więc $\pi_{N[t]}(x)=x$ oraz $\pi_{N[t]}(y)=y$, a zatem z~Definicji
\ref{def:contraction} wynika $f=\{x,y\}\in E(G)$. Każda krawędź ścieżki $P^\prime$ jest więc zarazem krawędzią grafu $G$, a~tym
samym $P^\prime$ jest ścieżką z~$u$ do~$w$ w~$G$.

Pokażemy, że $P^\prime$ jest najkrótsza również w~$G$. Gdyby istniała w~$G$ ścieżka $Q$ z~$u$ do~$w$ o~długości
$\len{Q}<\len{P^\prime}$, to na mocy Lematu~\ref{lem:projection-walk} w~$G^\prime$ istniałby spacer z~$u$ do~$w$ o~długości
co najwyżej $\len{Q}$. Z~Lematu~\ref{lem:walk-to-path} wynikałoby więc, że w~$G^\prime$ istnieje ścieżka z~$u$ do~$w$ o~długości
co najwyżej $\len{Q}<\len{P^\prime}$, sprzecznie z~wyborem $P^\prime$. Otrzymujemy
\[
  \len{P^\prime}=\dist{G}{u}{w}.
\]
Ponieważ $u,w\in T\setminus\{t\}$, z~Obserwacji~\ref{obs:shortest-paths} dostajemy krawędź $e\in E(P^\prime)\cap F$. Jej oba końce
leżą poza $N[t]$, więc $e/N[t]=e$ oraz $e\in F^\prime$. Zatem $E(P^\prime)\cap F^\prime\neq\emptyset$.

\medskip\noindent
\textbf{Przypadek 2: $t^\prime\in V(P^\prime)$.}
Jeżeli $t^\prime$ jest ostatnim wierzchołkiem ścieżki $P^\prime$, zastępujemy $P^\prime$ ścieżką przeciwną. Nie zmienia to ani długości,
ani zbioru krawędzi. Możemy więc zapisać
\[
  P^\prime=(y_0,\dots,y_r,y_{r+1},\dots,y_m),
  \qquad
  y_r=t^\prime,
\]
gdzie $r<m$. Oznaczmy przez
\[
  Q^\prime\coloneqq (t^\prime,y_{r+1},\dots,y_m)
\]
sufiks ścieżki $P^\prime$ zaczynający się w~$t^\prime$, oraz połóżmy $z\coloneqq y_m$. Wierzchołek $z$ jest końcem ścieżki
$P^\prime$ różnym od~$t^\prime$, zatem
\[
  z\in T^\prime\setminus\{t^\prime\}=T\setminus\{t\}.
\]
W~szczególności $z\notin N[t]$.

Jeżeli $r=0$, to $Q^\prime=P^\prime$, więc $Q^\prime$ jest ścieżką najkrótszą z~$t^\prime$ do~$z$ w~$G^\prime$. Jeżeli natomiast
$r>0$, to $Q^\prime$ jest podścieżką ścieżki najkrótszej $P^\prime$, a~zatem z~Lematu~\ref{lem:subpath}, zastosowanego w~grafie
$G^\prime$, wynika
\[
  \len{Q^\prime}=\dist{G^\prime}{t^\prime}{z}.
\]
W~obu sytuacjach możemy zastosować Lemat~\ref{lem:lifting-endpoint}. Dostajemy wierzchołek $v\in N(t)$ taki, że
\[
  Q=(t,v,y_{r+1},\dots,y_m)
\]
jest ścieżką najkrótszą z~$t$ do~$z$ w~$G$.

Ponieważ $t,z\in T$, z~Obserwacji~\ref{obs:shortest-paths} wynika istnienie krawędzi $e\in E(Q)\cap F$. Pierwsza krawędź ścieżki
$Q$ jest równa $\{t,v\}$, więc nie należy do~$F$, gdyż $t\notin V[F]$. Ponieważ $z\notin N[t]$, ścieżka $Q$ ma długość co
najmniej $2$, a~zatem $e\neq \{t,v\}$. Krawędź $e$ leży więc na odcinku $(v,y_{r+1},\dots,y_m)$.

Jeżeli $e=\{v,y_{r+1}\}$, to $\pi_{N[t]}(v)=t^\prime$ oraz $\pi_{N[t]}(y_{r+1})=y_{r+1}$, skąd
\[
  e/N[t]=\{t^\prime,y_{r+1}\},
\]
czyli $e/N[t]$ jest pierwszą krawędzią ścieżki $Q^\prime$. Jeżeli natomiast $e=\{y_i,y_{i+1}\}$ dla pewnego
$i\in\{r+1,\dots,m-1\}$, to oba końce $e$ leżą poza $N[t]$, więc $e/N[t]=e$, a~zatem $e/N[t]\in E(Q^\prime)$. W~obu
podprzypadkach mamy
\[
  e/N[t]\in E(Q^\prime)\subseteq E(P^\prime).
\]
Ponieważ $e\in F$, z~definicji zbioru $F^\prime$ dostajemy także $e/N[t]\in F^\prime$. Stąd ponownie
\[
  E(P^\prime)\cap F^\prime\neq\emptyset.
\]

W obu przypadkach każda ścieżka najkrótsza między różnymi wierzchołkami z~$T^\prime$ przecina zbiór $F^\prime$. Z~Obserwacji
\ref{obs:shortest-paths} wynika, że $F^\prime$ jest zbiorem na wpół-tnącym dla~$T^\prime$ w~$G^\prime$.
\end{proof}

\section{Główne twierdzenie}

\begin{theorem}\label{thm:main}
Niech $H$ będzie skończonym grafem nieskierowanym, niech $T\subseteq V(H)$ oraz $F\subseteq E(H)$. Jeżeli $F$ jest zbiorem
na wpół-tnącym dla~$T$ w~$H$, to
\[
  |F|\ge |T|-1.
\]
\end{theorem}

\begin{proof}
Niech $H^{\mathrm s}$ oraz $F^{\mathrm s}$ będą obiektami z~Lematu~\ref{lem:simple-reduction}. Jeżeli udowodnimy tezę dla grafu
prostego $H^{\mathrm s}$, to z~tego lematu otrzymamy
\[
  |T|-1\le |F^{\mathrm s}|\le |F|.
\]
Wystarczy więc rozpatrzyć przypadek, w~którym $H$ jest grafem prostym. Od tej chwili oznaczamy go przez $G=(V,E)$.

Dla par liczb całkowitych nieujemnych $(k,n)$ oznaczmy przez $\mathcal{P}(k,n)$ następujące zdanie:
\begin{quote}
dla każdego grafu prostego $H$ o~$n$ wierzchołkach, każdego zbioru $X\subseteq V(H)$ o~mocy $k$ oraz każdego zbioru krawędzi
$M\subseteq E(H)$, jeżeli $M$ jest zbiorem na wpół-tnącym dla~$X$ w~$H$, to $|M|\ge k-1$.
\end{quote}
Udowodnimy prawdziwość wszystkich zdań $\mathcal{P}(k,n)$ przez indukcję względem porządku leksykograficznego na parach $(k,n)$,
to znaczy przyjmujemy, że
\[
  (k_1,n_1)\prec (k_2,n_2)
  \quad\Longleftrightarrow\quad
  k_1<k_2
  \ \text{lub}\ 
  \bigl(k_1=k_2 \ \text{i}\ n_1<n_2\bigr).
\]
Porządek ten jest dobrze ufundowany na~$\mathbb{N}_0^2$, więc taka indukcja jest poprawna.

\medskip\noindent
\textbf{Baza indukcji.} Niech $k\le 1$. Wtedy dla dowolnego grafu $H$, dowolnego $X\subseteq V(H)$ o~mocy $k$ oraz dowolnego
$M\subseteq E(H)$ warunek z~Definicji~\ref{def:semi-cut} jest spełniony w~sposób pusty, ponieważ w~$X$ nie ma par różnych
wierzchołków. W~szczególności zachodzi nierówność
\[
  |M|\ge 0\ge k-1.
\]
Zatem $\mathcal{P}(k,n)$ jest prawdziwe dla każdego $n$.

\medskip\noindent
\textbf{Krok indukcyjny.} Ustalmy parę $(k,n)$ z~$k\ge 2$ i~załóżmy, że $\mathcal{P}(k',n')$ jest prawdziwe dla każdej pary
$(k',n')$ spełniającej $(k',n')\prec (k,n)$. Rozważmy graf $G=(V,E)$ o~$n$ wierzchołkach, zbiór $T\subseteq V$ o~mocy $k$
oraz zbiór $F\subseteq E$, który jest na wpół-tnący dla~$T$ w~$G$. Pokażemy, że $|F|\ge k-1$.

Rozpatrzymy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $T\cap V[F]=\emptyset$.}
Wybierzmy $t\in T$ oraz $u\in T\setminus\{t\}$. Z~Definicji~\ref{def:semi-cut} wynika, że $\dist{G}{t}{u}<\infty$. Istnieje więc
ścieżka z~$t$ do~$u$ o~dodatniej długości, a~zatem $N(t)\neq\emptyset$ i~w szczególności
\[
  |N[t]|\ge 2.
\]

Niech
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
Na mocy Propozycji~\ref{prop:contraction} zbiór $F^\prime$ jest na wpół-tnący dla~$T^\prime$ w~$G^\prime$. Ponadto
\[
  |T^\prime|=|T|=k
  \qquad\text{oraz}\qquad
  |V(G^\prime)|=|V|-|N[t]|+1\le n-1<n.
\]
Para odpowiadająca trójce $(G^\prime,T^\prime,F^\prime)$ jest więc równa $(k,n')$ dla pewnego $n'<n$, a~więc spełnia
$(k,n')\prec (k,n)$. Z~założenia indukcyjnego
otrzymujemy
\[
  |F^\prime|\ge |T^\prime|-1=k-1.
\]
Na mocy Lematu~\ref{lem:contraction-image-size} mamy $|F^\prime|\le |F|$, a~więc
\[
  |F|\ge k-1.
\]

\medskip\noindent
\textbf{Przypadek 2: $T\cap V[F]\neq\emptyset$.}
Wybierzmy wierzchołek $t\in T\cap V[F]$ oraz krawędź $e\in F$ taką, że $t\in e$. Oznaczmy
\[
  T^-\coloneqq T\setminus\{t\},
  \qquad
  F^-\coloneqq F\setminus\{e\}.
\]
Pokażemy, że $F^-$ jest zbiorem na wpół-tnącym dla~$T^-$ w~$G$.

Weźmy dowolne różne wierzchołki $u,w\in T^-$. Z~Definicji~\ref{def:semi-cut} wynika, że $\dist{G}{u}{w}<\infty$. Niech $P$ będzie
dowolną ścieżką najkrótszą z~$u$ do~$w$ w~$G$. Z~Obserwacji~\ref{obs:shortest-paths} dostajemy
\[
  E(P)\cap F\neq\emptyset.
\]
Jeżeli $e\notin E(P)$, to natychmiast
\[
  E(P)\cap F^-\neq\emptyset.
\]
Pozostaje przypadek $e\in E(P)$. Ponieważ $t\in e$, wierzchołek $t$ należy do~$V(P)$. Zapiszmy $P$ w~postaci
\[
  P=P_u\cdot P_w,
\]
gdzie $P_u$ jest podścieżką od~$u$ do~$t$, a~$P_w$ podścieżką od~$t$ do~$w$. Z~Lematu~\ref{lem:subpath} wynika, że
$P_u$ i~$P_w$ są ścieżkami najkrótszymi odpowiednio między parami $(u,t)$ i~$(t,w)$.

Ponieważ $P$ jest ścieżką, podścieżki $P_u$ i~$P_w$ mają wspólny tylko wierzchołek $t$, czyli
\[
  V(P_u)\cap V(P_w)=\{t\}.
\]
Gdyby krawędź $f$ należała jednocześnie do $E(P_u)$ i~$E(P_w)$, to oba jej końce należałyby do~$V(P_u)\cap V(P_w)=\{t\}$, co jest
niemożliwe, ponieważ każda krawędź grafu prostego ma dwa różne końce. Otrzymujemy zatem
\[
  E(P_u)\cap E(P_w)=\emptyset.
\]
Krawędź $e$ jest incydentna z~$t$, zatem należy dokładnie do~jednej z~tych dwóch podścieżek. Rozważamy dwa podprzypadki.

\smallskip\noindent
\textit{Podprzypadek 2a: $e\in E(P_u)$.}
Wtedy $e\notin E(P_w)$. Ponieważ $t,w\in T$ i~$P_w$ jest ścieżką najkrótszą z~$t$ do~$w$, z~Obserwacji~\ref{obs:shortest-paths}
wynika istnienie krawędzi $f\in E(P_w)\cap F$. Z~rozłączności zbiorów $E(P_u)$ i~$E(P_w)$ dostajemy $f\neq e$, a więc $f\in F^-$.
W szczególności
\[
  E(P)\cap F^-\neq\emptyset.
\]

\smallskip\noindent
\textit{Podprzypadek 2b: $e\in E(P_w)$.}
Wtedy $e\notin E(P_u)$. Ponieważ $u,t\in T$ i~$P_u$ jest ścieżką najkrótszą z~$u$ do~$t$, z~Obserwacji~\ref{obs:shortest-paths}
wynika istnienie krawędzi $f\in E(P_u)\cap F$. Z~rozłączności zbiorów $E(P_u)$ i~$E(P_w)$ dostajemy $f\neq e$, a więc $f\in F^-$.
Ponownie
\[
  E(P)\cap F^-\neq\emptyset.
\]

Wykazaliśmy, że każda ścieżka najkrótsza między różnymi wierzchołkami z~$T^-$ przecina zbiór $F^-$. Ponadto dla takich wierzchołków
odległość w~$G$ jest skończona, gdyż $T^-\subseteq T$ i~zbiór $F$ był na wpół-tnący dla~$T$. Z~Obserwacji~\ref{obs:shortest-paths}
wynika zatem, że $F^-$ jest zbiorem na wpół-tnącym dla~$T^-$ w~$G$.

Para odpowiadająca trójce $(G,T^-,F^-)$ jest równa $(k-1,n)$, a~więc spełnia $(k-1,n)\prec (k,n)$. Z~założenia
indukcyjnego otrzymujemy
\[
  |F^-|\ge |T^-|-1=(k-1)-1=k-2.
\]
Ponieważ $F^-=F\setminus\{e\}$ oraz $e\in F$, mamy
\[
  |F|=|F^-|+1\ge k-1.
\]
Krok indukcyjny jest zakończony, a tym samym prawdziwe jest zdanie $\mathcal{P}(k,n)$ dla każdej pary $(k,n)$.
\end{proof}

\begin{remark}
Oszacowanie z~Twierdzenia~\ref{thm:main} jest ostre. Jeżeli $G$ jest grafem będącym ścieżką oraz $T=V(G)$, to dla $F=E(G)$ zachodzi
$|F|=|V(G)|-1=|T|-1$, a~po usunięciu wszystkich krawędzi każda para różnych wierzchołków ma odległość równą $\infty$.
\end{remark}
