\section{Wprowadzenie}

Niech $G=(V,E)$ bńôdzie skoŇĄczonym grafem nieskierowanym prostym oraz niech $T\subseteq V$. Dla zbioru krawńôdzi

\begin{definition}[Spacery, ŇõcieŇľki i~odlegŇāoŇõńá]\label{def:paths-distance}

$\len{P}\coloneqq k$. ŇöcieŇľkńÖ nazywamy spacer, w~kt√≥rym wierzchoŇāki sńÖ parami r√≥Ňľne. Dla ŇõcieŇľki
$P=(v_0,\dots,v_k)$ piszemy $V(P)\coloneqq \{v_0,\dots,v_k\}$ oraz

\begin{notation}[Podstawowe operacje na grafie]\label{not:basic-operations}

\begin{definition}[ŇöcińÖganie niepustego zbioru wierzchoŇāk√≥w]\label{def:contraction}
Niech $\emptyset\neq W\subseteq V$ oraz niech $w^\ast\notin V$ bńôdzie nowym wierzchoŇākiem. Definiujemy odwzorowanie
\[
  \pi_W\colon V\to (V\setminus W)\cup\{w^\ast\},
  \qquad
  \pi_W(v)\coloneqq
  \begin{cases}
    w^\ast & \text{dla } v\in W,\\
    v & \text{dla } v\notin W.
  \end{cases}
\]
Przez $G/W$ oznaczamy graf prosty $(V',E')$, gdzie
\[
\begin{aligned}
  V' &\coloneqq (V\setminus W)\cup\{w^\ast\},\\
  E' &\coloneqq \bigl\{\{\pi_W(x),\pi_W(y)\}: \{x,y\}\in E,\ \pi_W(x)\neq \pi_W(y)\bigr\}.
\end{aligned}
\]
Warunek $\pi_W(x)\neq \pi_W(y)$ gwarantuje, Ňľe w~$G/W$ nie powstajńÖ pńôtle, a~opis zbiorowy krawńôdzi utoŇľsamia ewentualne
krawńôdzie r√≥wnolegŇāe; w~szczeg√≥lnoŇõci $G/W$ pozostaje grafem prostym.
JeŇľeli $e=\{x,y\}\in E$ oraz $\pi_W(x)\neq \pi_W(y)$, to obraz tej krawńôdzi oznaczamy przez
$e/W\coloneqq \{\pi_W(x),\pi_W(y)\}\in E'$.
Dla $F\subseteq E$ definiujemy obraz zbioru krawńôdzi przez
$F/W\coloneqq \bigl\{\{\pi_W(x),\pi_W(y)\}: \{x,y\}\in F,\ \pi_W(x)\neq \pi_W(y)\bigr\}\subseteq E'$.
\end{definition}

\begin{definition}[Zbi√≥r na wp√≥Ňā-tnńÖcy]\label{def:semi-cut}

\end{observation}

\begin{proof}

odlegŇāoŇõńá mińôdzy $u$ i~$w$ ŇõciŇõle roŇõnie. W~szczeg√≥lnoŇõci odlegŇāoŇõńá $\dist{G}{u}{w}$ jest skoŇĄczona. Istnienie skoŇĄczonej
odlegŇāoŇõci mińôdzy kaŇľdńÖ parńÖ terminali oznacza, Ňľe mińôdzy dowolnymi dwoma terminalami istnieje ŇõcieŇľka w~$G$. Zatem
wszystkie terminale naleŇľńÖ do jednej skŇāadowej sp√≥jnoŇõci grafu $G$.

\section{Lematy pomocnicze}

\begin{lemma}\label{lem:contraction-image-size}
Niech $\emptyset\neq W\subseteq V$ oraz $F\subseteq E$. W√≥wczas $|F/W|\le |F|$.
\end{lemma}

\begin{proof}

  \varphi(\{x,y\})\coloneqq \{\pi_W(x),\pi_W(y)\},

\end{proof}

\begin{lemma}\label{lem:walk-to-path}
Niech $H$ bńôdzie skoŇĄczonym grafem prostym. JeŇľeli w~$H$ istnieje spacer z~$u$ do~$w$, to istnieje r√≥wnieŇľ ŇõcieŇľka z~$u$ do~$w$

Jest to standardowa wŇāasnoŇõńá spacer√≥w: usuwajńÖc kolejne zamknińôte fragmenty mińôdzy dwoma wystńÖpieniami tego samego
wierzchoŇāka, otrzymujemy ŇõcieŇľkńô z~$u$ do~$w$ o~dŇāugoŇõci nie wińôkszej niŇľ dŇāugoŇõńá wyjŇõciowego spaceru; por.~\cite[rozdz.~1]{bondy-murty,diestel}.

\begin{lemma}\label{lem:subpath}

$\len{P}=\dist{H}{u}{w}$. W√≥wczas dla kaŇľdych indeks√≥w $0\le i\le j\le k$ zachodzi

\begin{proof}

z~$u$ do~$w$ kr√≥tszńÖ niŇľ $P$, wbrew r√≥wnoŇõci $\len{P}=\dist{H}{u}{w}$; por.~\cite[rozdz.~1]{bondy-murty,diestel}.
\end{proof}

\begin{observation}\label{obs:shortest-paths}
Niech $H$ bńôdzie skoŇĄczonym grafem prostym, niech $X\subseteq V(H)$ oraz niech $M\subseteq E(H)$. W√≥wczas $M$ jest zbiorem

ŇõcieŇľka $P$ z~$u$ do~$w$ o~dŇāugoŇõci $\len{P}=\dist{H}{u}{w}$ speŇānia $E(P)\cap M\neq\emptyset$.

\end{observation}

\begin{proof}

ZaŇā√≥Ňľmy najpierw, Ňľe $M$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$X$ w~$H$. Wtedy
$\dist{H}{u}{w}<\dist{H\setminus M}{u}{w}\le \infty$, wińôc $\dist{H}{u}{w}<\infty$. Niech $P$ bńôdzie ŇõcieŇľkńÖ z~$u$ do~$w$
o~dŇāugoŇõci $\dist{H}{u}{w}$. Gdyby $E(P)\cap M=\emptyset$,
to $P$ byŇāaby takŇľe ŇõcieŇľkńÖ w~$H\setminus M$, a~wińôc

ZaŇā√≥Ňľmy teraz, Ňľe $\dist{H}{u}{w}<\infty$ oraz kaŇľda ŇõcieŇľka z~$u$ do~$w$ o~dŇāugoŇõci $\dist{H}{u}{w}$ przecina zbi√≥r $M$.

\begin{lemma}\label{lem:projection-walk}

\end{proof}

na wp√≥Ňā-tnńÖcńÖ. UŇľyjemy do tego dw√≥ch mechanizm√≥w: projekcji ŇõcieŇľek do grafu ŇõcińÖgnińôtego oraz podnoszenia najkr√≥tszej

\begin{lemma}\label{lem:lifting-endpoint}
Niech $t\in V$, niech $t^\prime\notin V$ bńôdzie nowym wierzchoŇākiem oraz niech $G^\prime\coloneqq G/N[t]$, gdzie zbi√≥r
$N[t]$ zostaŇā ŇõcińÖgnińôty do~$t^\prime$. Niech $z\in V\setminus N[t]$ i~niech $P^\prime=(t^\prime,x_1,\dots,x_k)$ bńôdzie
ŇõcieŇľkńÖ najkr√≥tszńÖ z~$t^\prime$ do~$z$ w~$G^\prime$. W√≥wczas istnieje wierzchoŇāek $v\in N(t)$ taki, Ňľe
$P=(t,v,x_1,\dots,x_k)$ jest ŇõcieŇľkńÖ najkr√≥tszńÖ z~$t$ do~$z$ w~$G$.
\end{lemma}

\begin{proof}
PoniewaŇľ $z\neq t^\prime$, mamy $k\ge 1$. Z~r√≥wnoŇõci $V(G^\prime)=(V\setminus N[t])\cup\{t^\prime\}$ wynika, Ňľe wierzchoŇāek
$x_1$, jako r√≥Ňľny od~$t^\prime$, naleŇľy do~$V\setminus N[t]$.
KrawńôdŇļ $\{t^\prime,x_1\}$ naleŇľy do~$E(G^\prime)$, wińôc z~Definicji~\ref{def:contraction} istnieje pewien wierzchoŇāek
$v\in N[t]$ oraz odpowiadajńÖca mu krawńôdŇļ $\{v,x_1\}\in E$ taka, Ňľe $\{v,x_1\}/N[t]=\{t^\prime,x_1\}$.
Wybierzmy dowolny taki wierzchoŇāek $v$. Gdyby $v=t$, to z~$\{t,x_1\}\in E$ wynikaŇāoby $x_1\in N(t)\subseteq N[t]$, sprzecznie
z~$x_1\in V\setminus N[t]$. Zatem $v\in N(t)$.

PoniewaŇľ $v\in N(t)\subseteq N[t]$ oraz $x_1,\dots,x_k\in V\setminus N[t]$, Ňľaden z~wierzchoŇāk√≥w $x_1,\dots,x_k$ nie jest r√≥wny
ani~$t$, ani~$v$. Ponadto wierzchoŇāki $x_1,\dots,x_k$ sńÖ parami r√≥Ňľne, gdyŇľ $P^\prime$ jest ŇõcieŇľkńÖ. Dla kaŇľdego $i\in\{2,\dots,k\}$ krawńôdŇļ $\{x_{i-1},x_i\}$ naleŇľy
do~$E(G^\prime)$ i~ma oba koŇĄce poza $N[t]$, a~wińôc z~Definicji~\ref{def:contraction} wynika $\{x_{i-1},x_i\}\in E$. Zatem
$P=(t,v,x_1,\dots,x_k)$ jest ŇõcieŇľkńÖ w~$G$ i~speŇānia $\len{P}=\len{P^\prime}+1$. PoniewaŇľ $P^\prime$ jest ŇõcieŇľkńÖ najkr√≥tszńÖ,
mamy ponadto $\dist{G^\prime}{t^\prime}{z}=\len{P^\prime}<\infty$.

Niech teraz $Q=(q_0,\dots,q_\ell)$ bńôdzie dowolnńÖ ŇõcieŇľkńÖ z~$t$ do~$z$ w~$G$. Gdyby $\ell=1$, to z~$\{t,z\}\in E$
wynikaŇāoby $z\in N(t)\subseteq N[t]$, wbrew zaŇāoŇľeniu $z\notin N[t]$. Zatem $\ell\ge 2$ oraz
$\{q_0,q_1\}=\{t,q_1\}\in E$, skńÖd $q_1\in N(t)\subseteq N[t]$, a~zatem $\pi_{N[t]}(q_1)=t^\prime$. CińÖg $(q_1,\dots,q_\ell)$
jest koŇĄcowym fragmentem ŇõcieŇľki $Q$, a~wińôc spacerem w~$G$. Zastosowanie Lematu~\ref{lem:projection-walk} do tego spaceru
daje spacer z~$t^\prime$ do~$z$ w~$G^\prime$ o~dŇāugoŇõci co najwyŇľej $\ell-1$, gdyŇľ
$\pi_{N[t]}(q_\ell)=\pi_{N[t]}(z)=z$.
Z~Lematu~\ref{lem:walk-to-path} otrzymujemy wińôc ŇõcieŇľkńô z~$t^\prime$ do~$z$ w~$G^\prime$ o~dŇāugoŇõci co najwyŇľej $\ell-1$.
Wobec tego $\dist{G^\prime}{t^\prime}{z}\le \ell-1=\len{Q}-1$. PoniewaŇľ ostatnia nier√≥wnoŇõńá zachodzi dla kaŇľdej ŇõcieŇľki $Q$
z~$t$ do~$z$, wynika stńÖd $\dist{G}{t}{z}\ge \dist{G^\prime}{t^\prime}{z}+1$. Z~drugiej strony $P$ jest ŇõcieŇľkńÖ z~$t$ do~$z$,
wińôc $\dist{G}{t}{z}\le \len{P}=\len{P^\prime}+1=\dist{G^\prime}{t^\prime}{z}+1$. Obie nier√≥wnoŇõci dajńÖ
$\dist{G}{t}{z}=\len{P}$, czyli $P$ jest ŇõcieŇľkńÖ najkr√≥tszńÖ z~$t$ do~$z$.
\end{proof}

\begin{observation}\label{obs:outside-edges}
Niech $t\in V$, niech $t^\prime\notin V$ bńôdzie nowym wierzchoŇākiem oraz niech $G^\prime\coloneqq G/N[t]$. JeŇľeli
$x,y\in V\setminus N[t]$ sńÖ r√≥Ňľne, to $\{x,y\}\in E(G^\prime)$ wtedy i~tylko wtedy, gdy $\{x,y\}\in E(G)$, a~w tym przypadku
$\{x,y\}/N[t]=\{x,y\}$.
\end{observation}

\begin{proof}
PoniewaŇľ $x,y\notin N[t]$, mamy $\pi_{N[t]}(x)=x$ oraz $\pi_{N[t]}(y)=y$. JeŇľeli $\{x,y\}\in E(G)$, to z~Definicji~\ref{def:contraction}
wynika, Ňľe $\{x,y\}\in E(G^\prime)$ oraz $\{x,y\}/N[t]=\{x,y\}$. Odwrotnie, jeŇľeli $\{x,y\}\in E(G^\prime)$, to z~tej samej
definicji istnieje krawńôdŇļ $\{u,v\}\in E(G)$ taka, Ňľe $\{\pi_{N[t]}(u),\pi_{N[t]}(v)\}=\{x,y\}$.
PoniewaŇľ jedynym wierzchoŇākiem grafu $G^\prime$ leŇľńÖcym poza zbiorem $V\setminus N[t]$ jest $t^\prime$, z~r√≥wnoŇõci obraz√≥w
wynika, Ňľe $u,v\notin N[t]$. StńÖd $\pi_{N[t]}(u)=u$ oraz $\pi_{N[t]}(v)=v$, a~zatem $\{u,v\}=\{x,y\}\in E(G)$.
\end{proof}

\begin{proposition}\label{prop:contraction}
Niech $T\subseteq V$ oraz $F\subseteq E$. ZaŇā√≥Ňľmy, Ňľe $F$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$T$ w~$G$. Niech
$t\in T\setminus V[F]$ oraz niech $t^\prime\notin V$ bńôdzie nowym wierzchoŇākiem. Oznaczmy
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
W√≥wczas $F^\prime$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$T^\prime$ w~$G^\prime$.
\end{proposition}

\begin{proof}
Dow√≥d opiera sińô na dw√≥ch mechanizmach. Krawńôdzie o~obu koŇĄcach poza $N[t]$ przechodzńÖ do $G^\prime$ bez zmiany
na mocy Obserwacji~\ref{obs:outside-edges}. JeŇľeli natomiast ŇõcieŇľka najkr√≥tsza w~$G^\prime$ przechodzi przez $t^\prime$,
to jej koŇĄcowy fragment wychodzńÖcy z~$t^\prime$ moŇľna, na mocy Lematu~\ref{lem:lifting-endpoint}, podnieŇõńá do ŇõcieŇľki
najkr√≥tszej wychodzńÖcej z~$t$ w~grafie $G$. Najpierw pokaŇľemy, Ňľe $T\setminus\{t\}\subseteq V\setminus N[t]$.
WeŇļmy dowolne $u\in T\setminus\{t\}$ i~przypuŇõńámy, Ňľe $u\in N(t)$. Wtedy $\dist{G}{t}{u}=1$, a~jedyna ŇõcieŇľka dŇāugoŇõci $1$
z~$t$ do~$u$ jest r√≥wna $(t,u)$. Z~Obserwacji~\ref{obs:shortest-paths} wynikaŇāoby wińôc, Ňľe krawńôdŇļ $\{t,u\}$ naleŇľy do~$F$,
co przeczy zaŇāoŇľeniu $t\notin V[F]$. Otrzymana sprzecznoŇõńá dowodzi ŇľńÖdanej inkluzji. W~szczeg√≥lnoŇõci $T^\prime\subseteq V(G^\prime)$.

W~nastńôpnym kroku pokaŇľemy, Ňľe dla kaŇľdych r√≥Ňľnych $u,w\in T^\prime$ zachodzi $\dist{G^\prime}{u}{w}<\infty$. JeŇľeli
$u,w\in T\setminus\{t\}$, to z~Obserwacji~\ref{obs:finite-terminal-distances} mamy $\dist{G}{u}{w}<\infty$. Istnieje wińôc ŇõcieŇľka z~$u$ do~$w$
w~$G$. PoniewaŇľ $u,w\notin N[t]$, mamy $\pi_{N[t]}(u)=u$ oraz $\pi_{N[t]}(w)=w$, wińôc z~Lemat√≥w~\ref{lem:projection-walk}
i~\ref{lem:walk-to-path} otrzymujemy ŇõcieŇľkńô z~$u$ do~$w$ w~$G^\prime$. JeŇľeli jeden z~wierzchoŇāk√≥w $u,w$ jest r√≥wny $t^\prime$, to drugi naleŇľy do~$T\setminus\{t\}$;
oznaczmy go przez $z$. Z~Obserwacji~\ref{obs:finite-terminal-distances} mamy $\dist{G}{t}{z}<\infty$, wińôc w~$G$ istnieje
ŇõcieŇľka z~$t$ do~$z$.
PoniewaŇľ $\pi_{N[t]}(t)=t^\prime$ oraz $\pi_{N[t]}(z)=z$, z~Lematu~\ref{lem:projection-walk}, zastosowanego do tej ŇõcieŇľki,
otrzymujemy spacer z~$t^\prime$ do~$z$ w~$G^\prime$, a~nastńôpnie z~Lematu~\ref{lem:walk-to-path} ŇõcieŇľkńô z~$t^\prime$ do~$z$
w~$G^\prime$. Tym samym kaŇľda para r√≥Ňľnych terminali z~$T^\prime$
ma w~$G^\prime$ skoŇĄczonńÖ odlegŇāoŇõńá.

Niech teraz $u,w\in T^\prime$ bńôdńÖ r√≥Ňľne i~niech $P^\prime$ bńôdzie dowolnńÖ ŇõcieŇľkńÖ najkr√≥tszńÖ z~$u$ do~$w$ w~$G^\prime$.
PoniewaŇľ na mocy Definicji~\ref{def:contraction} graf $G^\prime$ jest grafem prostym, moŇľemy zastosowańá do niego
Obserwacjńô~\ref{obs:shortest-paths}. Wystarczy wińôc wykazańá, Ňľe $E(P^\prime)\cap F^\prime\neq\emptyset$.
Rozpatrujemy dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: $t^\prime\notin V(P^\prime)$.}
Wtedy $V(P^\prime)\subseteq V\setminus N[t]$,
bo jedynym wierzchoŇākiem grafu $G^\prime$ spoza zbioru $V\setminus N[t]$ jest wŇāaŇõnie $t^\prime$. KaŇľda krawńôdŇļ ŇõcieŇľki
$P^\prime$ ma wińôc oba koŇĄce poza $N[t]$, a~zatem na mocy Obserwacji~\ref{obs:outside-edges} jest zarazem krawńôdzińÖ grafu
$G$. Wynika stńÖd, Ňľe $P^\prime$ jest ŇõcieŇľkńÖ z~$u$ do~$w$ r√≥wnieŇľ w~$G$. PoniewaŇľ koŇĄce ŇõcieŇľki naleŇľńÖ do~$V(P^\prime)$,
mamy takŇľe $u,w\neq t^\prime$, a~zatem $u,w\in T^\prime\setminus\{t^\prime\}=T\setminus\{t\}$.

  \qquad

\end{proof}

\begin{proof}
Udowodnimy nieco silniejsze sformuŇāowanie: dla kaŇľdej pary liczb nieujemnych $(k,n)$, kaŇľdego grafu prostego $H$ o~$n$
wierzchoŇākach, kaŇľdego zbioru terminali $X\subseteq V(H)$ o~mocy $k$ oraz kaŇľdego zbioru krawńôdzi $M\subseteq E(H)$, jeŇľeli
$M$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$X$ w~$H$, to zachodzi nier√≥wnoŇõńá $|M|\ge k-1$. W~naszym zastosowaniu wystńôpujńÖ wyŇāńÖcznie
pary $(k,n)$ speŇāniajńÖce $k\le n$, gdyŇľ zawsze mamy $X\subseteq V(H)$. Dow√≥d prowadzimy przez indukcjńô wzglńôdem porzńÖdku
leksykograficznego
\[
  (k_1,n_1)\prec (k_2,n_2)
  \quad\Longleftrightarrow\quad
  k_1<k_2
  \ \text{lub}\
  \bigl(k_1=k_2\ \text{i}\ n_1<n_2\bigr).
\]
PorzńÖdek ten jest dobrze ufundowany na zbiorze
\[
  \bigl\{(a,b)\in\mathbb{N}_0^2:\ a\le b\bigr\}.
\]
Sama indukcja po~$|T|$ nie wystarcza, poniewaŇľ operacja ŇõcińÖgania zachowuje liczbńô terminali i~zmniejsza jedynie liczbńô
wierzchoŇāk√≥w.

\medskip\noindent
\textbf{Baza indukcji.} Niech $k\le 1$. Wtedy zbi√≥r par r√≥Ňľnych wierzchoŇāk√≥w z~$X$ jest pusty, wińôc
$|M|\ge 0\ge k-1$, co koŇĄczy bazńô indukcji.

\medskip\noindent
\textbf{Krok indukcyjny.} Ustalmy parńô $(k,n)$ z~$k\ge 2$ i~zaŇā√≥Ňľmy, Ňľe rozwaŇľana teza jest prawdziwa dla kaŇľdej pary
$(k',n')\prec (k,n)$. RozwaŇľmy graf prosty $G=(V,E)$ o~$n$ wierzchoŇākach, zbi√≥r terminali $T\subseteq V$ o~mocy $k$ oraz zbi√≥r
krawńôdzi $F\subseteq E$, kt√≥ry jest na wp√≥Ňā-tnńÖcy dla~$T$ w~$G$. PokaŇľemy, Ňľe
$|F|\ge k-1$.
JeŇľeli znajdziemy terminal naleŇľńÖcy do~$T\setminus V[F]$, przejdziemy do mniejszego grafu przez ŇõcińÖgnińôcie jego
domknińôtego otoczenia. JeŇľeli natomiast $T\subseteq V[F]$, zachowamy graf, lecz usuniemy jeden terminal i~jednńÖ krawńôdŇļ
ze zbioru $F$. Rozpatrujemy wińôc dwa przypadki.

\medskip\noindent
\textbf{Przypadek 1: istnieje terminal $t\in T\setminus V[F]$.}
PoniewaŇľ $k\ge 2$, moŇľemy wybrańá wierzchoŇāek $u\in T\setminus\{t\}$. Z~Obserwacji~\ref{obs:finite-terminal-distances} mamy
$\dist{G}{t}{u}<\infty$. Istnieje wińôc ŇõcieŇľka z~$t$ do~$u$. PoniewaŇľ $u\neq t$, ŇõcieŇľka ta ma dodatnińÖ dŇāugoŇõńá, a~zatem jej
pierwsza krawńôdŇļ jest incydentna z~$t$. W~szczeg√≥lnoŇõci wierzchoŇāek $t$ ma co najmniej jednego sńÖsiada, czyli
$N(t)\neq\emptyset$ oraz $|N[t]|\ge 2$.

Niech $t^\prime\notin V$ bńôdzie nowym wierzchoŇākiem i~niech
\[
  G^\prime\coloneqq G/N[t],
  \qquad
  T^\prime\coloneqq (T\setminus\{t\})\cup\{t^\prime\},
  \qquad
  F^\prime\coloneqq F/N[t].
\]
Sytuacjńô tńô ilustruje Rysunek~\ref{fig:main-contraction}.
PoniewaŇľ z~zaŇāoŇľenia tego przypadku mamy $t\in T\setminus V[F]$, na mocy Propozycji~\ref{prop:contraction} zbi√≥r $F^\prime$
jest zbiorem na wp√≥Ňā-tnńÖcym dla~$T^\prime$ w~$G^\prime$. PoniewaŇľ
$t^\prime\notin V$ oraz $T\subseteq V$, mamy $t^\prime\notin T$, a~wińôc $|T^\prime|=|T|=k$. Ponadto
$|V(G^\prime)|=|V|-|N[t]|+1\le n-1<n$. Z~inkluzji $T^\prime\subseteq V(G^\prime)$ wynika takŇľe
$|T^\prime|=k\le |V(G^\prime)|$. Skoro $|T^\prime|=k$ oraz $|V(G^\prime)|<n$, zachodzi zatem ostra nier√≥wnoŇõńá leksykograficzna
$(|T^\prime|,|V(G^\prime)|)\prec (k,n)$. Z~zaŇāoŇľenia indukcyjnego otrzymujemy
$|F^\prime|\ge |T^\prime|-1=k-1$. Lemat~\ref{lem:contraction-image-size} daje $|F^\prime|\le |F|$, a~zatem $|F|\ge k-1$.

\medskip\noindent
\textbf{Przypadek 2: $T\subseteq V[F]$.}
PoniewaŇľ $T\subseteq V[F]$, kaŇľdy terminal jest incydentny z~co najmniej jednńÖ krawńôdzińÖ ze zbioru $F$. Wybierzmy wińôc
$t\in T$ oraz krawńôdŇļ $e\in F$ incydentnńÖ z~$t$. Oznaczmy $T^-\coloneqq T\setminus\{t\}$ oraz $F^-\coloneqq F\setminus\{e\}$.
PokaŇľemy, Ňľe $F^-$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$T^-$ w~$G$.

WeŇļmy dowolne r√≥Ňľne wierzchoŇāki $u,w\in T^-$. PoniewaŇľ $T^-\subseteq T$ i~$F$ jest zbiorem na wp√≥Ňā-tnńÖcym dla~$T$, z~Obserwacji~\ref{obs:finite-terminal-distances} mamy
$\dist{G}{u}{w}<\infty$. Niech $P$ bńôdzie dowolnńÖ ŇõcieŇľkńÖ najkr√≥tszńÖ z~$u$ do~$w$ w~$G$. Z~Obserwacji~\ref{obs:shortest-paths}
dostajemy $E(P)\cap F\neq\emptyset$. JeŇľeli $e\notin E(P)$, to od razu $E(P)\cap F^-\neq\emptyset$.
Pozostaje przypadek $e\in E(P)$.

PoniewaŇľ $e$ jest incydentna z~$t$, istnieje wierzchoŇāek $s\in V$ taki, Ňľe $e=\{t,s\}$. Z~zaŇāoŇľenia $e\in E(P)$ wynika
wińôc, Ňľe oba koŇĄce krawńôdzi $e$ naleŇľńÖ do~$V(P)$, a~w szczeg√≥lnoŇõci $t\in V(P)$. Zapiszmy $P=(v_0,\dots,v_m)$. Istnieje
dokŇāadnie jeden indeks $j\in\{0,\dots,m\}$ taki, Ňľe $v_j=t$, gdyŇľ $P$ jest ŇõcieŇľkńÖ. Definiujemy
$P_u\coloneqq (v_0,\dots,v_j)$ oraz $P_w\coloneqq (v_j,\dots,v_m)$. PoniewaŇľ $v_0=u\neq t$ oraz $v_m=w\neq t$, zachodzi
$0<j<m$. W√≥wczas $P_u$ jest podŇõcieŇľkńÖ od~$u$ do~$t$, $P_w$ jest podŇõcieŇľkńÖ od~$t$ do~$w$, a~ponadto $P=P_u\cdot P_w$.
Na mocy Lematu~\ref{lem:subpath} $P_u$ i~$P_w$ sńÖ ŇõcieŇľkami najkr√≥tszymi odpowiednio mińôdzy parami $(u,t)$ i~$(t,w)$.
Sytuacjńô tńô ilustruje
Rysunek~\ref{fig:split-path}.
StńÖd obie te podŇõcieŇľki majńÖ dodatnińÖ dŇāugoŇõńá.

PoniewaŇľ $P$ jest ŇõcieŇľkńÖ, podŇõcieŇľki $P_u$ i~$P_w$ majńÖ wsp√≥lny jedynie wierzchoŇāek $t$. W~szczeg√≥lnoŇõci
\[
  E(P)=E(P_u)\cup E(P_w)
  \qquad\text{oraz}\qquad
  E(P_u)\cap E(P_w)=\emptyset,
\]
gdyŇľ gdyby krawńôdŇļ $g$ naleŇľaŇāa do~$E(P_u)\cap E(P_w)$, to oba jej koŇĄce naleŇľaŇāyby do~zbioru
\[
  V(P_u)\cap V(P_w)=\{t\},
\]
a~wińôc mielibyŇõmy $g=\{t,t\}$, co jest niemoŇľliwe w~grafie prostym. KrawńôdŇļ $e$ naleŇľy wińôc dokŇāadnie do jednej z~tych
dw√≥ch podŇõcieŇľek.

JeŇľeli $e\in E(P_u)$, to z~rozŇāńÖcznoŇõci zbior√≥w $E(P_u)$ i~$E(P_w)$ mamy $e\notin E(P_w)$. PoniewaŇľ $t,w\in T$ oraz $P_w$
